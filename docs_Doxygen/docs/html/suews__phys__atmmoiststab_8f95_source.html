247  FUNCTION stab_fn_mom(StabilityMethod, ZL, zl_f) RESULT(psym)
248  ! StabilityMethod = 1-4 -
249  ! PSYM - stability FUNCTION for momentum
250  !Modified by LJ Mar 2010
251  !Input:Used stability method, stability (z-d)/L, zeta (either (z-d)/L or z0/L)
252 
253  ! USE mod_z
254  ! USE mod_k
255 
256  IMPLICIT NONE
257  REAL(kind(1d0)), PARAMETER :: &
258  ! k=0.4,& !Von Karman's contant
259  ! k2=0.16,& !Power of Van Karman's contant
260  neut_limit = 0.001000 !Limit for neutral stability
261  ! notUsedI=-55
262 
263  REAL(kind(1d0)):: piover2, psym, zl, zl_f, x, x2
264  INTEGER ::stabilitymethod
265 
266  piover2 = acos(-1.)/2.
267  !PRINT*,StabilityMethod,zl,"stab_fn_mom:"
268  IF (abs(zl) < neut_limit) THEN
269  psym = 0
270  ELSEIF (zl < -neut_limit) THEN !Unstable
271 
272  IF (stabilitymethod == 1) THEN ! Jensen et al 1984 - Van Ulden & Holtslag (1985) p 1206&
273  psym = ((1.-16.*zl_f)**0.25) - 1
274  ELSEIF (stabilitymethod == 2) THEN !Dyer (1974)(1-16z/L)**.25' k=0.41 mod. Hogstrom (1988)v15.2
275  x = (1.-(15.2*zl_f))**0.25
276  x2 = log((1 + (x**2.))/2.)
277  psym = (2.*log((1 + x)/2.)) + x2 - (2.*atan(x)) + piover2
278  ELSEIF (stabilitymethod == 3) THEN ! Kondo (1975) adopted by Campbell & Norman eqn 7.26 p 97
279  psym = 0.6*(2)*log((1 + (1 - 16*zl_f)**0.5)/2)
280  ELSEIF (stabilitymethod == 4) THEN !Businger et al (1971) modifed Hogstrom (1988)
281  x = (1 - 19.3*zl_f)**(-0.25)
282  x2 = log((1 + (x**2.))/2.)
283  psym = (2.*log((1 + x)/2.)) + x2 - (2.*atan(x)) + piover2
284  ELSEIF (stabilitymethod == 7) THEN ! Dyer & Bradley (1982) (1-28z/L)**.25' k=0.4
285  x = (1 - (28.*zl_f))**0.25 ! NT: changed + to - (bug -> checked reference)
286  x2 = log((1 + x**2.)/2.)
287  psym = (2.*log((1 + x)/2.)) + x2 - (2.*atan(x)) + piover2
288  ELSEIF (stabilitymethod == 5) THEN ! Zilitinkevich & Chalikov (1968) modified Hogstrom (1988)
289  IF (zl_f >= -0.16) THEN
290  x = 1 + 1.38*zl_f
291  ELSE
292  x = 0.42*(-1)*zl_f**0.333
293  ENDIF
294  x2 = log((1 + (x**2.))/2.)
295  psym = (2.*log((1 + x)/2.)) + x2 - (2.*atan(x)) + piover2
296 
297  ELSEIF (stabilitymethod == 6) THEN ! Foken and Skeib (1983)
298  IF (zl_f >= 0.06) THEN
299  x = 1
300  ELSE
301  x = ((-1)*zl_f/0.06)**0.25
302  ENDIF
303  x2 = log((1 + (x**2.))/2.)
304  psym = (2.*log((1 + x)/2.)) + x2 - (2.*atan(x)) + piover2
305  ENDIF
306 
307  ELSEIF (zl > neut_limit) THEN !Stable
308 
309  IF (stabilitymethod == 1) THEN !Dyer (1974) k=0.35 x=1+5*zl Mod. Hogstrom (1988)
310  psym = (-4.8)*zl_f
311  ELSEIF (stabilitymethod == 2) THEN !Van Ulden & Holtslag (1985) p 1206
312  IF (zl_f > 1000.) THEN
313  zl_f = 1000.
314  END IF
315  psym = (-17.*(1.-exp(-0.29*zl_f)))
316  ELSEIF (stabilitymethod == 4) THEN ! Businger et al (1971) modifed Hogstrom (1988)
317  ! psym=1+6*zl_f ! this is NOT the integral form but the stability function, TS 13 Jun 2017
318  psym = (-6)*zl_f ! this is the integral form, TS 13 Jun 2017
319  ELSEIF (stabilitymethod == 3) THEN ! Kondo (1975) adopted by Campbell & Norman eqn 7.26 p 97
320  psym = (-6)*log(1 + zl_f)
321 
322  ENDIF
323  ENDIF
324  RETURN
325  END FUNCTION stab_fn_mom

401  psyh = 0
402  ELSEIF (zl < -neut_limit) THEN ! Unstable
403  IF (stabilitymethod == 3) THEN
404  !campbell & norman eqn 7.26
405  psyh = 0.6*(2)*log((1 + (1 - 16*zl_f)**0.5)/2)
406  ELSE
407

409  x = 0.95*(1.-11.6*zl_f)**(-0.5)
410  ELSEIF (stabilitymethod == 7) THEN
411  x = (1 - (28.*zl))**0.25

413  x = 0.95*(1.-15.2*zl_f)**0.5
414  ENDIF
415  psyh = 2*log((1 + x**2)/2)
416  ENDIF
417 
418  ELSE IF (zl > neut_limit) THEN !Stable
419  IF (zl <= 1) THEN ! weak/moderate stable
420  IF (stabilitymethod == 4) THEN !Businger et al (1971) modifed Hogstrom (1988)
421  ! psyh=0.95+(7.8*zl_f) ! this is NOT the integral form but the stability function, TS 13 Jun 2017
422  psyh = (-7.8)*zl_f ! this is the integral form, TS 13 Jun 2017
423  ELSE !Dyer (1974) PSYH=(-5)*ZL modifed Hogstrom (1988)
424  psyh = (-4.5)*zl_f
425  ENDIF
426  ELSE !zL>1, very stable. otherwise psyh would be too large. TS 13 Jun 2017
427  ! adopt the form as Brutasert (1982) eqn 4.58. but following the coeffs. of the above eqns
428  IF (stabilitymethod == 4) THEN !Businger et al (1971) modifed Hogstrom (1988)
429  psyh = (-7.8)*(1 + log(zl_f))
430  ELSE !Dyer (1974) PSYH=(-5)*ZL modifed Hogstrom (1988)
431  psyh = (-4.5)*(1 + log(zl_f))
432  ENDIF
433  END IF
434 
435  ENDIF

437  RETURN
