26  FUNCTION sat_vap_press(TK, P) RESULT(es)
27  !c sg sept 99 f90
28  !c This uses eqns from Buck (1981) JAM 20, 1527-1532
29  !c units T (K) e (mb) P (mb)
30  !c f corrects for the fact that we are not dealing with pure water
31  REAL(kind(1d0)) :: tk, p, tc, es, e, f
32  tc = tk - 273.15
33  IF (tc == 0) THEN
34  tc = 0.001
35  ENDIF
36  !Valid for 50>T>-40
37  e = 6.1121*exp(((18.729 - tc/227.3)*tc)/(tc + 257.87))
38  f = 1.00072 + p*(3.2e-6 + 5.9e-10*tc**2)
39  es = e*f
40  END FUNCTION sat_vap_press

274  FUNCTION lat_vap(Temp_C, Ea_hPa, Press_hPa, cp, dectime) RESULT(lv_J_kg)
275  !Input: Air temperature, Water vapour pressure, Air pressure, heat capacity
276  !Output: latent heat of vaporization
277 
278  ! USE time
279  ! USE SnowMod
280  ! USE defaultnotUsed
281 
282  IMPLICIT NONE
283  REAL(kind(1d0))::cp, lv_j_kg, ea_fix, tw, &
284  incr, es_tw, psyc, ea_est, press_hpa, ea_hpa, temp_c, dectime!,Temp_K
285  ! REAL(KIND(1d0))::sat_vap_press,psyc_const ! functions
286 
287  LOGICAL:: switch1 = .false., switch2 = .false.!,debug=.true.
288  INTEGER:: ii, from = 2
289  REAL(kind(1d0)), PARAMETER::notused = -55.55
290 
291  ea_fix = ea_hpa
292  !if(debug) write(*,*)Temp_C, 'LV'
293  !Temp_K=temp_C+273.16
294 
295  !lv=1.91846E6*(Temp_K/(Temp_K-33.91))**2
296 
297  lv_j_kg = (2500.25 - 2.365*temp_c)*1000 !First guess for lv in units J/kg
298 
299  tw = temp_c/2. !First estimate for wet bulb temperature
300  incr = 3.
301  DO ii = 1, 100
302  IF (press_hpa < 900) THEN
303  CALL errorhint(45, 'function Lat_vap', press_hpa, notused, ii)
304  ENDIF

310  !if(debug.and.dectime>55.13.and.dectime<55.2)write(35,*)'% 2',Tw

319  CALL errorhint(45, 'function Lat _vap -31', press_hpa, notused, ii)

321

323 
324  lv_j_kg = (2500.25 - 2.365*tw)*1e3
325 
326  IF (switch1 .AND. switch2) THEN
327  incr = incr/10.
328  switch1 = .false.

333  ELSEIF (ea_est > ea_fix) THEN

336  ELSEIF (ea_est < ea_fix) THEN
337  tw = tw + incr
338  switch2 = .true.
339  ENDIF
340  ENDDO
341 
342  RETURN
343  END FUNCTION lat_vap

493  ! MRR, 1987
494  ! AT TEMPERATURE T (DEG C) AND PRESSURE PMB (MB), GET SATURATION SPECIFIC

496

498 
499  REAL(kind(1d0)), PARAMETER:: &
500  !Teten coefficients
501  a = 6.106, &
502  b = 17.27, &
503  c = 237.3, &
504  molar = 0.028965, & !Dry air molar fraction in kg/mol
505  molar_wat_vap = 0.0180153 !Molar fraction of water vapor in kg/mol
506 
507  IF (t > 55) THEN
508  CALL errorhint(34, 'Function qsatf', t, 0.00d0, -55)
509  ENDIF
510
