72  ! More water cannot drain than is in the surface state_id

656 
657  INTEGER::jj, is
658  REAL(KIND(1d0)):: &

660  SoilMoistCap_Vol1, &
661  SoilMoist_vol1, &
662  SoilMoistCap_Vol2, &
663  SoilMoist_vol2, &
664  B_r1, MatPot1, Km1, &
665  B_r2, MatPot2, Km2, &
666  Distance, KmWeight, dI, &
667  dI_dt!Water flow between two stores
668 
669  REAL(KIND(1d0)), PARAMETER:: &
670  alphavG = 0.0005, & !Set alphavG to match value in van Genuchten (1980) [mm-1]
671  nunits = 1 !Can change to represent plot/base unit size
672 
673  ! SoilMoist_vol1,2 = Volumetric soil moisture [m3 m-3]
674  ! SoilMoistCap_vol1,2 = Volumetric soil moisture capacity [m3 m-3] (from FunctionalTypes)
675  ! MatPot1,2 = Water potential (i.e. pressure head) of store [mm]
676  ! DimenWaterCon1,2 = Dimensionless water content, or relative saturation [-]
677  ! Distance = Distance between two stores [m]
678  ! B_r1,2 = Residual volumetric soil moisture [m3 m-3]
679  ! Km1,2 = Hydraulic conductivity of store [mm s-1]
680  ! KmWeight = Weighted hydraulic conductivity [mm s-1]
681  ! alphavG = Parameter (could depend on soil texture) [mm-1]
682  ! dI = Water flow between stores [mm] dI = dI_dt * no. secs in each timestep
683  ! if dI > 0, first surface gains water, second surface loses water
684  ! NUnits = Number of repeating units (e.g. properties, blocks) for distance calculation [-]

687

689

691  ! and is capable of storing water (SoilStoreCap [mm])
692  DO jj = is + 1, nsurf - 1 !is-1,1,-1 !Sub-loop through remaining surfaces (runs backwards as of 13/08/2014, HCW)
693 
694  IF (sfr(jj) /= 0 .AND. soilstorecap(jj) > 0) THEN !If other surface area exists
695  ! and is capable of storing water
696 
697  ! ---- For surface 1 -----------------------------------------------------
698  ! Calculate non-saturated VWC
699  soilmoistcap_vol1 = soilstorecap(is)/soildepth(is) !Volumetric soil moisture capacity [m3 m-3] (i.e. saturated VWC)
700  soilmoist_vol1 = soilstore_id(is)/soildepth(is) !Volumetric soil moisture [m3 m-3]
701 
702  !B_r1=SoilMoistCap_Vol1-SoilMoist_vol1 !Residual soil moisture content [m3 m-3]
703  b_r1 = 0.1 !HCW 12/08/2014 Temporary fix
704  ! Need to add residual soil moisture values to FunctionalTypes
705  !B_r1=VolSoilMoistRes(is) !Residual soil moisture content [m3 m-3]
706 
707  !Order of if statements reversed HCW 22 Feb 2017
708  !If soil moisture less than or equal to residual value, set MatPot to max and Km to 0 to suppress water movement
709  IF (b_r1 >= soilmoist_vol1) THEN
710  matpot1 = 100000
711  km1 = 0 !Added by LJ in Nov 2013
712  ! Otherwise, there should be enough water in the soil to allow horizontal transfer
713  ELSE
714  dimenwatercon1 = (soilmoist_vol1 - b_r1)/(soilmoistcap_vol1 - b_r1) !Dimensionless water content [-]
715 
716  ! If very large or very small, adjust for calculation of MatPot and Km
717  IF (dimenwatercon1 > 0.99999) THEN
718  dimenwatercon1 = dimenwatercon1 - 0.0001 !This cannot equal 1
719  ENDIF
720 
721  IF (dimenwatercon1 < 0.00000005) THEN
722  dimenwatercon1 = dimenwatercon1 + 0.0000001 !Added HCW 22 Feb 2017
723  ENDIF
724 
725  !van Genuchten (1980), with n=2 and m = 1-1/n = 1/2
726  !Water potential of first store [mm] (van Genuchten 1980, Eq 3 rearranged)
727  matpot1 = sqrt(1/dimenwatercon1**2 - 1)/alphavg
728 
729  !Hydraulic conductivity of first store [mm s-1] (van Genuchten 1980, Eq 8)
730  km1 = sathydraulicconduct(is)*sqrt(dimenwatercon1)*(1 - (1 - dimenwatercon1**2)**0.5)**2
731 
732  !Check this value (HCW 12/08/2014)
733  IF (matpot1 > 100000) THEN
734  matpot1 = 100000 !Max. potential is 100000 mm (van Genuchten 1980)
735  km1 = 0 !Added by HCW 12/08/2014
736  ENDIF
737 
738  ENDIF
739 
740  ! ---- For surface 2 -----------------------------------------------------
741  ! Calculate non-saturated VWC
742  soilmoistcap_vol2 = soilstorecap(jj)/soildepth(jj) !Volumetric soil moisture capacity [m3 m-3] (i.e. saturated VWC)
743  soilmoist_vol2 = soilstore_id(jj)/soildepth(jj) !Volumetric soil moisture [m3 m-3]
744 
745  !B_r2=SoilMoistCap_Vol2-SoilMoist_vol2 !Residual soil moisture content [m3 m-3]
746  b_r2 = 0.1 !HCW 12/08/2014 Temporary fix
747  ! Need to add residual soil moisture values to FunctionalTypes
748  !B_r2=VolSoilMoistRes(jj) !Residual soil moisture content [m3 m-3]
749 
750  !If soil moisture below residual value, set MatPot to maximum
751  IF (b_r2 >= soilmoist_vol2) THEN
752  matpot2 = 100000
753  km2 = 0 !Added by LJ in Nov 2013
754  ELSE
755  dimenwatercon2 = (soilmoist_vol2 - b_r2)/(soilmoistcap_vol2 - b_r2) !Dimensionless water content [-]
756 
757  IF (dimenwatercon2 > 0.99999) THEN
758  dimenwatercon2 = dimenwatercon2 - 0.0001 !This cannot equal 1
759  ENDIF
760 
761  IF (dimenwatercon2 < 0.00000005) THEN

766  !Water potential of second store [mm] (van Genuchten 1980, Eq 3 rearranged)
767  matpot2 = sqrt(1/dimenwatercon2**2 - 1)/alphavg
768 
769  !Hydraulic conductivity of second store [mm s-1] (van Genuchten 1980, Eq 8)
770  km2 = sathydraulicconduct(jj)*sqrt(dimenwatercon2)*(1 - (1 - dimenwatercon2**2)**0.5)**2
771 
772  IF ((matpot2) > 100000) THEN
773  matpot2 = 100000 !Max. potential is 100000 mm (van Genuchten 1980)

777  ENDIF

784

786  kmweight = (sfr(is)*km1 + sfr(jj)*km2)/(sfr(is) + sfr(jj))
787

789  !Multiply Distance by 1000 to convert m to mm (HCW 12/08/2014)
790  di_dt = -(kmweight)*(-matpot1 + matpot2)/(distance*1000)
791

793  !Added by HCW 12/08/2014
794  di = di_dt*tstep_real !Use dI instead of dI_dt in the following calculations
795

798

805  ELSEIF ((soilstore_id(is) + di) < 0) THEN

808 
809  ! If insufficient water in second surface to move dI, instead move as much as possible
810  ELSE
811  soilstore_id(is) = soilstore_id(is) + soilstore_id(jj)*sfr(jj)/sfr(is)
812  soilstore_id(jj) = 0
813  ENDIF
814 
815  !If soil moisture exceeds capacity, excess goes to soil runoff (first surface)
816  IF (soilstore_id(is) > soilstorecap(is)) THEN
817  runoffsoil(is) = runoffsoil(is) + (soilstore_id(is) - soilstorecap(is))
818  soilstore_id(is) = soilstorecap(is)
819  !elseif (soilstore_id(is)<0) then !HCW 13/08/2014 commented out as should never be true here anyway...
820  ! soilstore_id(is)=0 ! ... and if so, need to do more here (i.e. account for other water too)
821  ENDIF
822 
823  !If soil moisture exceeds capacity, excess goes to soil runoff (second surface)
824  IF (soilstore_id(jj) > soilstorecap(jj)) THEN
825  runoffsoil(jj) = runoffsoil(jj) + (soilstore_id(jj) - soilstorecap(jj))
826  soilstore_id(jj) = soilstorecap(jj)
827  !elseif (soilstore_id(jj)<0) then !HCW 13/08/2014 commented out (as above)
828  ! soilstore_id(jj)=0
829  ENDIF
830 
831  ENDIF !end if second surface exists and is capable of storing water
832 
833  ENDDO !end jj loop over second surface
834 
835  runoffsoil_per_tstep = runoffsoil_per_tstep + (runoffsoil(is)*sfr(is)/nonwaterfraction) !Excludes water body. Moved here as otherwise code crashed when NonWaterFraction=0
836 
837  ENDIF !end if first surface exists and is capable of storing water
838 
839  !runoffSoil_per_tstep=runoffSoil_per_tstep+(runoffSoil(is)*sfr(is)/NonWaterFraction) !Excludes water body
840 
841  ENDDO !is loop over first surface
842
