547  SUBROUTINE rkutta(neqn, XA, XB, Y, NSTEPS)
548  ! XA=s0
549  ! XB=s1
550  ! Y(1)=blh_m
551  ! Y(2)=tm_K
552  ! Y(3)=qm_kgkg
553  ! Y(4)=cm
554  ! Y(5)=tp_K
555  ! Y(6)=qp_kgkg
556  ! JOHN KNIGHT, 1985 (AMENDED BY MRR, 23-SEP-85)
557  ! EXPLICIT FOURTH-ORDER RUNGE-KUTTA METHOD FOR FIRST-ORDER ODE SYSTEM
558  ! OF NE EQUATIONS, WITH INITIAL VALUES SUPPLIED.
559  ! MEANING OF PARAMETERS:
560  ! NE = NUMBER OF EQUATIONS (MAX 21)
561  ! XA = VALUE OF INDEPENDENT VARIABLE ON ENTRY
562  ! XB = VALUE OF INDEPENDENT VARIABLE AT END OF INTERVAL
563  ! Y(NE) = ON ENTRY: INITIAL VALUES OF DEPENDENT VARIABLES AT XA
564  ! ON EXIT: CALCULATED VALUES OF DEPENDENT VARIABLES AT XB
565  ! NSTEPS = NUMBER OF INTEGRATION STEPS OVER INTERVAL (XA,XB)
566  ! DIFF = NAME OF USER-SUPPLIED SUBROUTINE TO CALCULATE DERIVATIVES
567  ! DYDX (DIFF MUST BE DECLARED EXTERNAL IN CALLING PROGRAM).
568  ! PARAMETERS IN SUBROUTINE DIFF(NE,X,Y,DYDX):
569  ! NEqn = NUMBER OF EQUATIONS
570  ! X = INDEPENDENT VARIABLE
571  ! Y = ARRAY (LENGTH NE) OF VALUES OF DEPENDENT VARIABLES
572  ! DYDX = ON EXIT, ARRAY (LENGTH NE) OF VALUES OF DERIVATIVES
573  ! IMPLICIT real*8 (A-H,O-Z)
574  IMPLICIT NONE
575  INTEGER::ns, nsteps, nj, n, neqn
576  REAL(KIND(1D0)), DIMENSION(neqn):: y
577  REAL(KIND(1D0)), DIMENSION(21):: dydx, arg
578  REAL(KIND(1D0)), DIMENSION(21, 5):: rk
579  REAL(KIND(1D0)), DIMENSION(4):: coef
580  REAL(KIND(1D0)):: XA, XB, step, X, xx
581 
582  coef(1) = 1.0
583  coef(2) = 0.5
584  coef(3) = 0.5
585  coef(4) = 1.0
586  ! print*,"rk1: ",xa,xb,y
587  step = (xb - xa)/nsteps
588 
589  DO ns = 1, nsteps
590  DO nj = 1, neqn
591  rk(nj, 1) = 0
592  ENDDO
593  x = xa + (ns - 1)*step
594  DO n = 1, 4
595  IF (n == 1) THEN
596  xx = x
597  ELSEIF (n > 1) THEN
598  xx = x + coef(n)*step
599  ENDIF
600 
601  DO nj = 1, neqn
602  arg(nj) = y(nj) + coef(n)*rk(nj, n)
603  ENDDO
604 
605  CALL diff(xx, arg, dydx)
606 
607  DO nj = 1, neqn
608  rk(nj, n + 1) = step*dydx(nj)
609  ENDDO
610  ENDDO
611 
612  DO nj = 1, neqn
613  DO n = 1, 4
614  y(nj) = y(nj) + rk(nj, n + 1)/(6*coef(n))
615  ENDDO
616  ENDDO
617  ENDDO
618 
619  RETURN
620  END SUBROUTINE rkutta

624  SUBROUTINE diff(s, y1, dyds)
625  ! in y1,neqn
626  ! out dyds
627 
628  ! calculates derivatives for cbl slab model
629  ! y(1) = h = cbl depth(m)
630  ! y(2) = t = potential temp(K)
631  ! y(3) = q = specific humidity(kg/kg)
632  ! y(4) = c = CO2 concentration
633  USE data_in
634  USE sues_data
635  ! use allocateArray
636  USE time
637  USE cbl_module
638  USE defaultnotused
639  USE mod_grav
640 
641  IMPLICIT NONE
642  REAL(KIND(1D0)), DIMENSION(neqn)::dyds, y1
643  REAL(KIND(1d0)) :: zero = 0.0
644  REAL(KIND(1d0)) :: h1, t_K, q_kgkg, c, cp, ws, s, foo
645  ! real(kind(1D0)) :: tp_K,qp_kgkg
646  REAL(KIND(1D0)):: delt_K, delq_kgkg, delc
647  REAL(KIND(1D0)):: gamtv_Km, deltv_K, ftv_Kms
648  REAL(KIND(1D0)):: ftva_Kms, delb, qs2, qs3
649  REAL(KIND(1D0)):: dhds, dtds, dqds, dcds, dtpds, dqpds
650  REAL(KIND(1D0)):: conk, conn, cona, conc, cont
651 
652  ! print*,"diff: timestamp:",s
653  foo = s
654  ! pause
655  h1 = y1(1)!m
656  t_k = y1(2)!K
657  q_kgkg = y1(3)!kg/kg
658  c = y1(4)
659  tp_k = y1(5)!K
660  qp_kgkg = y1(6)!kg/kg
661 
662  ! find t, q, c above inversion, and jumps across inversion
663  ! tp = tp + gamt*h
664  ! qp = qp0 + gamq*h
665 
666  cp = 0 ! cp0 + gamc* h1 ! todo
667 
668  delt_k = tp_k - t_k
669  delq_kgkg = qp_kgkg - q_kgkg
670  delc = cp - c
671 
672  ! find potential virtual temperature flux, gradient and jump
673  ftv_kms = fhbl_kms + 0.61*tm_k*febl_kgkgms
674  gamtv_km = gamt_km + 0.61*tm_k*gamq_kgkgm!/1000
675  deltv_k = delt_k + 0.61*tm_k*delq_kgkg
676 
677  ! find velocity scale ws
678  ftva_kms = max(ftv_kms, zero) ! virtual heat flux
679  ws = (h1*ftva_kms*grav/tm_k)**0.3333333333
680 
681  ! find dhds using one of 4 alternative schemes chosen by ient:
682  IF (entrainmenttype == 2) THEN
683  ! EntrainmentType=1: encroachment (as in McN and S 1986 eq 16))
684  dhds = ftva_kms/(h1*gamtv_km)
685 
686  ELSE IF (entrainmenttype == 1) THEN
687  ! EntrainmentType=2: Driedonks 1981 (as in McN and S 1986 eq 13)
688  IF (deltv_k <= 0.01) THEN
689  dhds = ftva_kms/(h1*gamtv_km)
690  CALL errorhint(30, "subroutine diff [CBL: Deltv_K<0.01 EntrainmentType=1], deltv_K,delt_K,", deltv_k, delt_k, notusedi)
691  CALL errorhint(30, "subroutine diff [CBL: Deltv_K<0.01 EntrainmentType=1], tm_K,TPP_K,y1", tm_k, tpp_k, notusedi)
692  ! call errorHint(31,"subroutine diff [CBL: Deltv_K<0.01 EntrainmentType=1], y1",real(y1(1),kind(1d0)),notUsed,notUsedI)
693  ELSE
694  delb = grav*deltv_k/tm_k
695  conc = 0.2
696  cona = 5.0
697  dhds = (conc*ws**3 + cona*cbldata(8)**3)/(h1*delb)
698  END IF
699 
700  ELSE IF (entrainmenttype == 4) THEN
701  ! EntrainmentType=3: Tennekes 1973 (as in R 1991 eqs 3,4)
702  alpha3 = 0.2 ! alpha changed back to original Tennekes 1973 value
703  IF (deltv_k <= 0.01) THEN
704  dhds = ftva_kms/(h1*gamtv_km)
705  CALL errorhint(31, 'subroutine difflfnout: [CBL: deltv_K<0.01 EntrainmentType=4],deltv_K', &
706  deltv_k, notused, notusedi)
707  ELSE
708  ! include the option whether or not to include subsidence
709  IF (isubs == 1) THEN
710  dhds = alpha3*ftva_kms/deltv_k + wsb
711  ELSE
712  dhds = alpha3*ftva_kms/deltv_k
713  END IF
714  END IF
715 
716  ! write (4,*) tpp, gamq, dhds, deltv
717 
718  ELSE IF (entrainmenttype == 3) THEN
719  ! EntrainmentType=4: Rayner and Watson 1991 eq 21
720  conn = 1.33
721  conk = 0.18
722  cont = 0.80
723  qs3 = ws**3 + (conn*cbldata(8))**3
724  qs2 = qs3**(0.6666666667)
725 
726  IF (deltv_k <= 0.01) THEN
727  dhds = ftva_kms/(h1*gamtv_km)
728  CALL errorhint(31, 'subroutine difflfnout: [CBL: deltv_K<0.01 EntrainmentType=3],deltv_K', &
729  deltv_k, notused, notusedi)
730 
731  ELSE
732  delb = grav*deltv_k/tm_k
733  dhds = (conk*qs3)/(cont*qs2 + h1*delb)
734  END IF
735 
736  ELSE
737  CALL errorhint(24, 'BLUEWS_DIff- CBL- illegal alpha', notused, notused, notusedi)
738  END IF
739  ! find dtds, dqds, dc/ds:
740  ! wsb is the subsidence velocity. Try using: -0.01, -0.05, -0.1.
741  IF (isubs == 1) THEN
742  dtds = fhbl_kms/h1 + delt_k*(dhds - wsb)/h1
743  dqds = febl_kgkgms/h1 + delq_kgkg*(dhds - wsb)/h1
744  dcds = fcbl/h1 + delc*(dhds - wsb)/h1
745  ! also iterate the top of CBL scalars
746  dtpds = gamt_km*(dhds - wsb)
747  dqpds = gamq_kgkgm*(dhds - wsb)
748  ELSE
749  dtds = fhbl_kms/h1 + delt_k*(dhds)/h1
750  dqds = febl_kgkgms/h1 + delq_kgkg*(dhds)/h1
751  dcds = fcbl/h1 + delc*(dhds)/h1
752  ! also iterate the top of CBL scalars
753  dtpds = gamt_km*(dhds)
754  dqpds = gamq_kgkgm*(dhds)
755  END IF
756 
757  dyds(1) = dhds
758  dyds(2) = dtds
759  dyds(3) = dqds
760  dyds(4) = dcds
761  dyds(5) = dtpds
762  dyds(6) = dqpds
763 
764  RETURN
765  END SUBROUTINE diff
