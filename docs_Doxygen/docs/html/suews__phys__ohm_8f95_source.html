191 SUBROUTINE ohm_coef_cal(sfr, nsurf, &
192  Tair_mav_5d, OHM_coef, OHM_threshSW, OHM_threshWD, &

194  BldgSurf, WaterSurf, &

200  SnowUse, & ! option for snow related calculations
201  BldgSurf, WaterSurf ! code for specific surfaces
202  REAL(KIND(1d0)), INTENT(in) :: &
203  sfr(nsurf), & ! surface cover fractions
204  SnowFrac(nsurf), & ! snow fractions of each surface
205  Tair_mav_5d, & ! Tair_mav_5d=HDD(id-1,4) HDD at the begining of today (id-1)
206  OHM_coef(nsurf + 1, 4, 3), &
207  OHM_threshSW(nsurf + 1), OHM_threshWD(nsurf + 1), & ! OHM thresholds
208  soilstore_id(nsurf), & ! soil moisture

210  state_id(nsurf) ! wetness status

212 
213  REAL(KIND(1d0)) :: surfrac
214  INTEGER :: i, ii, is
215 
216  ! OHM coefficients --------
217  ! Set to zero initially
218  a1 = 0 ![-]
219  a2 = 0 ![h]
220  a3 = 0 ![W m-2]
221  ! -------------------------
222 
223  ! Loop through surface types ----------------------------------------------------------
224  DO is = 1, nsurf
225  surfrac = sfr(is)
226 
227  ! Use 5-day running mean Tair to decide whether it is summer or winter ----------------
228  IF (tair_mav_5d >= ohm_threshsw(is)) THEN !Summer
229  ii = 0
230  ELSE !Winter
231  ii = 2
232  ENDIF
233 
234  IF (state_id(is) > 0) THEN !Wet surface
235  i = ii + 1
236  ELSE !Dry surface
237  i = ii + 2
238  ! If the surface is dry but SM is close to capacity, use coefficients for wet surfaces
239  IF (is > bldgsurf .AND. is /= watersurf) THEN !Wet soil (i.e. EveTr, DecTr, Grass, BSoil surfaces)
240  IF (soilstore_id(is)/soilstorecap(is) > ohm_threshwd(is)) THEN
241  i = ii + 1
242  ENDIF
243  ENDIF
244  ENDIF
245 
246  ! If snow, adjust surface fractions accordingly
247  IF (snowuse == 1 .AND. is /= bldgsurf .AND. is /= watersurf) THEN ! QUESTION: Why is BldgSurf excluded here?
248  surfrac = surfrac*(1 - snowfrac(is))
249  ENDIF
250 
251  ! Calculate the areally-weighted OHM coefficients
252  a1 = a1 + surfrac*ohm_coef(is, i, 1)
253  a2 = a2 + surfrac*ohm_coef(is, i, 2)
254  a3 = a3 + surfrac*ohm_coef(is, i, 3)
255 
256  ENDDO !end of loop over surface types ------------------------------------------------
257 END SUBROUTINE ohm_coef_cal

260 SUBROUTINE ohm_dqndt_cal_x(dt, dt_since_start, qn1_av, qn1, dqndt)
261  IMPLICIT NONE
262  INTEGER, INTENT(in) :: dt ! time step [s]
263  INTEGER, INTENT(in) :: dt_since_start ! time since simulation starts [s]
264  REAL(KIND(1d0)), INTENT(in) :: qn1 ! new qn1 value [W m-2]
265  REAL(KIND(1d0)), INTENT(inout) :: qn1_av ! weighted average of qn1 [W m-2]
266  REAL(KIND(1d0)), INTENT(inout) :: dqndt ! dQ* per dt for 60 min [W m-2 h-1]
267  REAL(KIND(1d0)), PARAMETER :: dt0_thresh = 3600 ! threshold for period of dqndt0 [s]
268  REAL(KIND(1d0)), PARAMETER :: window_hr = 2 ! window size for Difference calculation [hr]
269 
270  INTEGER :: dt0 ! period of dqndt0 [s]
271 
272  REAL(KIND(1d0)) :: qn1_av_0 !, qn1_av_start,qn1_av_end
273 
274  ! if previous period shorter than dt0_thresh, expand the storage/memory period
275  IF (dt_since_start < dt0_thresh) THEN ! spinup period
276  dt0 = dt_since_start + dt
277 
278  ELSE ! effective period
279  dt0 = dt0_thresh
280  ENDIF
281 
282  ! get weighted average at a previous time specified by `window_hr`
283  qn1_av_0 = qn1_av - dqndt*(window_hr - dt/3600.)
284 
285  ! averaged qn1 for previous period = dt0_thresh
286  qn1_av = (qn1_av*(dt0 - dt) + qn1*dt)/(dt0)
287 
288  ! do weighted average to calculate the difference by using the memory value and new forcing value
289  ! NB: keep the output dqndt in [W m-2 h-1]
290  dqndt = (qn1_av - qn1_av_0)/window_hr
291 
292 END SUBROUTINE ohm_dqndt_cal_x

306  dqndt = -999 ! initialise as -999

308  ! Store instantaneous qn1 values for previous hour (qn1_store_grid) and average (qn1_av)

310  qn1_store_grid = cshift(qn1_store_grid, 1) ! shift to left with one place
311  qn1_store_grid(nsh) = qn1
312  nsh_nna = count(qn1_store_grid /= -999, dim=1) !Find how many are not -999s !bug fixed HCW 08 Feb 2017

314  ELSEIF (nsh == 1) THEN
315  qn1_store_grid(:) = qn1
316  qn1_av = qn1
317  ENDIF
318  ! Store hourly average values (calculated every timestep) for previous 2 hours
319  IF (nsh > 1) THEN
320  qn1_av_store_grid = cshift(qn1_av_store_grid, 1)
321  qn1_av_store_grid(2*nsh + 1) = qn1_av
322  ELSEIF (nsh == 1) THEN
323  qn1_av_store_grid(:) = qn1_av
324  ENDIF
325  ! Calculate dQ* per dt for 60 min (using running mean Q* at t hours and (t-2) hours)
326  IF (any(qn1_av_store_grid == -999)) THEN
327  dqndt = 0 ! Set dqndt term to zero for spinup
328  ELSE
329  dqndt = 0.5*(qn1_av_store_grid((2*nsh + 1)) - qn1_av_store_grid(1))
