227 SUBROUTINE daylen(DOY, XLAT, DAYL, DEC, SNDN, SNUP)
228  !-----------------------------------------------------------------------
229  IMPLICIT NONE
230  INTEGER :: DOY
231  REAL(KIND(1d0)) :: DEC, DAYL, SOC, SNDN, SNUP, XLAT
232  REAL(KIND(1d0)), PARAMETER :: PI = 3.14159, rad = pi/180.0
233 
234  !-----------------------------------------------------------------------
235  ! Calculation of declination of sun (Eqn. 16). Amplitude= +/-23.45
236  ! deg. Minimum = DOY 355 (DEC 21), maximum = DOY 172.5 (JUN 21/22).
237  dec = -23.45*cos(2.0*pi*(doy + 10.0)/365.0)
238 
239  ! Sun angles. SOC limited for latitudes above polar circles.
240  soc = tan(rad*dec)*tan(rad*xlat)
241  soc = min(max(soc, -1.0), 1.0)
242 
243  ! Calculate daylength, sunrise and sunset (Eqn. 17)
244  dayl = 12.0 + 24.0*asin(soc)/pi
245  snup = 12.0 - dayl/2.0
246  sndn = 12.0 + dayl/2.0
247 
248 END SUBROUTINE daylen
