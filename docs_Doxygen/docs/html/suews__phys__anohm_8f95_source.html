359  SUBROUTINE anohm_coef_land_cal( &
360  ASd, mSd, ATa, mTa, tau, mWS, mWF, mAH, & ! input: forcing
361  xalb, xemis, xcp, xk, xch, xBo, & ! input: sfc properties
362  xa1, xa2, xa3, ATs, mTs, gamma) ! output
363 
364  IMPLICIT NONE
365 
366  ! input: forcing scales
367  REAL(KIND(1d0)), INTENT(in):: ASd
368  REAL(KIND(1d0)), INTENT(in):: mSd
369  REAL(KIND(1d0)), INTENT(in):: ATa
370  REAL(KIND(1d0)), INTENT(in):: mTa
371  REAL(KIND(1d0)), INTENT(in):: tau
372  REAL(KIND(1d0)), INTENT(in):: mWS
373  REAL(KIND(1d0)), INTENT(in):: mWF
374  REAL(KIND(1d0)), INTENT(in):: mAH
375  ! input: sfc properties
376  REAL(KIND(1d0)), INTENT(in):: xalb
377  REAL(KIND(1d0)), INTENT(in):: xemis
378  REAL(KIND(1d0)), INTENT(in):: xcp
379  REAL(KIND(1d0)), INTENT(in):: xk
380  REAL(KIND(1d0)), INTENT(in):: xch
381  REAL(KIND(1d0)), INTENT(in):: xBo
382 
383  ! output
384  REAL(KIND(1d0)), INTENT(out) :: xa1
385  REAL(KIND(1d0)), INTENT(out) :: xa2
386  REAL(KIND(1d0)), INTENT(out) :: xa3
387  REAL(KIND(1d0)), INTENT(out) :: ATs
388  REAL(KIND(1d0)), INTENT(out) :: mTs
389  REAL(KIND(1d0)), INTENT(out) :: gamma
390 
391  ! constant
392  REAL(KIND(1d0)), PARAMETER :: SIGMA = 5.67e-8 ! Stefan-Boltzman
393  REAL(KIND(1d0)), PARAMETER :: PI = atan(1.0)*4 ! Pi
394  REAL(KIND(1d0)), PARAMETER :: OMEGA = 2*pi/(24*60*60) ! augular velocity of Earth
395 
396  ! local variables:
397  REAL(KIND(1d0)) :: beta ! inverse Bowen ratio
398  REAL(KIND(1d0)) :: f, fL, fT ! energy redistribution factors
399  REAL(KIND(1d0)) :: lambda ! thermal diffusivity
400  REAL(KIND(1d0)) :: delta, m, n ! water flux related variables
401  REAL(KIND(1d0)) :: ms, ns ! m, n related
402  REAL(KIND(1d0)) :: ceta, cphi ! phase related temporary variables
403  REAL(KIND(1d0)) :: eta, phi, xlag ! phase related temporary variables
404  REAL(KIND(1d0)) :: xx1, xx2, xx3, xchWS ! temporary use
405  ! LOGICAL :: flagGood = .TRUE. ! quality flag, T for good, F for bad
406 
407  ! give fixed values for test
408  ! properties
409  ! xalb = .2
410  ! xemis = .9
411  ! xcp = 1e6
412  ! xk = 1.2
413  ! xch = 4
414  ! xBo = 2
415  ! forcings
416  ! ASd = 400
417  ! mSd = 100
418  ! ATa = 23
419  ! mTa = 23+C2K
420  ! tau = PI/6
421  ! WS = 1
422  ! AH = 0
423  ! Wf = 0
424 
425  ! PRINT*, '********sfc_typ: ',sfc_typ,' start********'
426  ! initialize flagGood as .TRUE.
427  ! flagGood = .TRUE.
428  ! PRINT*, flagGood
429 
430  ! calculate sfc properties related parameters:
431  xchws = xch*mws
432 
433  ! xch = CRA/RA !update bulk trsf. coeff. with RA (aerodyn. res.)
434  IF (abs(xbo) < 0.001) THEN
435  beta = 1/0.001
436  ELSE
437  beta = 1/xbo
438  END IF
439 
440  ! PRINT*, 'beta:', beta
441  f = ((1 + beta)*xchws + 4*sigma*xemis*mta**3)
442  ! print*, 'xch',xch,'mTa',mTa,'dLu',4*SIGMA*xemis*mTa**3
443  fl = 4*sigma*xemis*mta**3
444  ! print*, 'fL',fL
445  ft = (1 + beta)*xchws
446  ! print*, 'fT',fT
447  lambda = xk/xcp
448  ! print*, 'lambda',lambda
449  delta = sqrt(.5*(mwf**2 + sqrt(mwf**4 + 16*lambda**2*omega**2)))
450  ! print*, 'delta',delta
451  m = (2*lambda)/(delta + mwf)
452  n = delta/omega
453  ! print*, 'm',m,'n',n
454 
455  ! calculate surface temperature related parameters:
456  mts = (msd*(1 - xalb)/f) + mta
457  ms = 1 + xk/(f*m)
458  ns = xk/(f*n)
459  ! print*, 'ms,ns:', ms,ns
460  xx1 = f*sin(tau)*ata
461  ! print*, 'xx1',xx1
462  xx2 = (1 - xalb)*asd + f*cos(tau)*ata
463  ! print*, 'xx2',xx2
464  gamma = atan(ns/ms) + atan(xx1/xx2)
465  ! print*, 'gamma:', gamma
466  ats = -(sin(tau)*ata)/(ns*cos(gamma) - ms*sin(gamma))
467  ! print*, 'ATs:', ATs
468 
469  ! calculate net radiation related parameters:
470  xx1 = (ns*cos(gamma) + sin(gamma) - ms*sin(gamma))*sin(tau)*ata*fl
471  xx2 = (xalb - 1)*(ns*cos(gamma) - ms*sin(gamma))*asd
472  xx3 = (-ms*cos(tau)*sin(tau) + cos(gamma)*(ns*cos(tau) + sin(tau)))*ata*fl
473  xx2 = xx2 - xx3
474  phi = atan(xx1/xx2)
475  xx3 = (ns*cos(gamma) - ms*sin(gamma))
476  xx1 = (1 + sin(gamma)/xx3)*sin(tau)*ata*fl
477  xx2 = (xalb - 1)*asd - (cos(tau) + cos(gamma)*sin(tau)/xx3)*ata*fl
478  cphi = sqrt(xx1**2 + xx2**2)
479 
480  ! calculate heat storage related parameters:
481  xx1 = m*cos(gamma) - n*sin(gamma)
482  xx2 = m*sin(gamma) + n*cos(gamma)
483  eta = atan(xx1/xx2)
484  ! if ( eta<0 ) print*, 'lambda,delta,m,n,gamma:', lambda,delta,m,n,gamma
485  xx1 = xk**2*(m**2 + n**2)*ats**2
486  xx2 = m**2*n**2
487  ceta = sqrt(xx1/xx2)
488 
489  ! key phase lag:
490  xlag = eta - phi
491 
492  ! calculate the OHM coeffs.:
493  ! a1:
494  xa1 = (ceta*cos(xlag))/cphi
495 
496  ! a2:
497  xa2 = (ceta*sin(xlag))/(omega*cphi)
498  xa2 = xa2/3600 ! convert the unit from s-1 to h-1
499 
500  ! a3:
501  xa3 = -xa1*(ft/f)*(msd*(1 - xalb)) - mah
502 
503  ! print*, 'ceta,cphi', ceta,cphi
504  ! print*, 'tau,eta,phi,xlag in deg:',tau/pi*180,eta/pi*180,phi/pi*180,xlag/pi*180
505 
506  ! PRINT*, '********sfc_typ: ',sfc_typ,' end********'
507 
508  END SUBROUTINE anohm_coef_land_cal

530  REAL(KIND(1d0)), INTENT(in):: mWF
531  REAL(KIND(1d0)), INTENT(in):: mAH

533  ! input: sfc properties
534  REAL(KIND(1d0)), INTENT(in):: xalb
535  REAL(KIND(1d0)), INTENT(in):: xemis

552  REAL(KIND(1d0)), PARAMETER :: SIGMA = 5.67e-8 ! Stefan-Boltzman
553  REAL(KIND(1d0)), PARAMETER :: PI = atan(1.0)*4 ! Pi
554  REAL(KIND(1d0)), PARAMETER :: OMEGA = 2*pi/(24*60*60) ! augular velocity of Earth

557  REAL(KIND(1d0)) :: beta ! inverse Bowen ratio

559  REAL(KIND(1d0)) :: lambda, calb ! temporary use
560  REAL(KIND(1d0)) :: delta ! sfc. temperature related variables
561  ! REAL(KIND(1d0)) :: delta,m,n ! sfc. temperature related variables
562  REAL(KIND(1d0)) :: xm, xn ! m, n related

567  REAL(KIND(1d0)) :: zeta, theta, xlag ! phase related temporary variables
568  REAL(KIND(1d0)) :: xx1, xx2, xx3 ! temporary use
569  REAL(KIND(1d0)) :: kappa ! temporary use
570  REAL(KIND(1d0)) :: dtau, dpsi, dphi ! temporary use
571  REAL(KIND(1d0)) :: cdtau, cdpsi, cdphi ! temporary use

580  ! calculate sfc properties related parameters:
581  xm = xk*xmu**2
582  xn = xcp*omega
583  phi = atan(xn/xm)
584  kappa = sqrt(xcp*omega/(2*xk))
585 
586  ! mWS=SUM(WS, dim=1, mask=(WS>0))/SIZE(WS, dim=1)
587  xchws = xch*mws
588  beta = 1/xbo
589  f = ((1 + beta)*xchws + 4*sigma*xemis*mta**3)
590  fl = 4*sigma*xemis*mta**3
591  ft = (1 + beta)*xchws
592 
593  calb = 1 - xalb
594 
595  lambda = sqrt(xm**2 + xn**2)
596 
597  dtau = atan(xk*kappa/(f + xk*kappa))
598  dpsi = atan((xk - xmu)/kappa)
599  dphi = atan(kappa*(f + xk*xmu)/(f*(kappa - xmu) + xk*kappa*(2*kappa - xmu)))
600  cdtau = sqrt((xk*kappa)**2 + (f + xk*kappa)**2)
601  cdpsi = sqrt((xk - xmu)**2 + kappa**2)
602  cdphi = cdtau*cdpsi
603 
604  ! calculate surface temperature related parameters:
605  ! daily mean:
606  mts = (msd*(1 - xalb + xeta)/f) + mta
607  ! amplitude:
608  xx1 = (xk*xeta*xmu*calb*asd*cdpsi)**2
609  xx2 = 2*lambda*sqrt(xx1)*(calb*asd*sin(phi - dpsi) + f*ata*sin(tau + phi - dpsi))
610  xx3 = lambda**2*((calb*asd + cos(tau)*f*ata)**2 + (sin(tau)*f*ata)**2)
611  ats = 1/(cdtau*lambda)*sqrt(xx1 + xx2 + xx3)
612  ! phase lag:
613  xx1 = (xk*kappa*calb*asd + cdtau*f*ata*sin(tau + dtau))*lambda &
614  + (xk*xeta*xmu*calb*asd*cdphi*sin(phi + dphi))
615  xx2 = ((f + xk*kappa)*calb*asd - cdtau*f*ata*cos(tau + dtau))*lambda &
616  - xk*xeta*xmu*calb*asd*cdphi*cos(phi + dphi)
617  delta = atan(xx1/xx2)
618  gamma = delta
619 
620  ! calculate net radiation related parameters:
621  ! phase lag:
622  xx1 = fl*(ats*sin(delta) - ata*sin(tau))
623  xx2 = calb*asd - fl*(ats*cos(delta) - ata*cos(tau))
624  theta = atan(xx1/xx2)
625  ! amplitude:
626  ctheta = sqrt(xx1**2 + xx2**2)
627 
628  ! calculate heat storage related parameters:
629  ! scales:
630  xxt = sqrt(2.)*kappa*lambda*ats
631  xxkappa = cdpsi*xeta*xmu*asd
632  xxdltphi = cos(delta)*sin(dpsi)*cos(phi) - sin(delta)*cos(dpsi)*sin(phi)
633  ! phase lag:
634  xx1 = xxt*sin(pi/4 - delta) + xxkappa*sin(phi + dpsi)
635  xx2 = xxt*sin(pi/4 + delta) - xxkappa*sin(phi - dpsi)
636  zeta = atan(xx1/xx2)
637  ! amplitude:
638  xx1 = 2*sqrt(2.)*xxkappa*xxt*xxdltphi
639  xx2 = (1 - cos(2*dpsi)*cos(2*phi))*xxkappa**2
640  xx3 = xxt**2
641  czeta = xk/lambda*sqrt(xx1 + xx2 + xx3)
642 
643  ! calculate the OHM coeffs.:
644  xlag = zeta - theta
645  ! a1:
646  xa1 = (czeta*cos(xlag))/ctheta
647 
648  ! write(*,*) 'ceta,xlag,cphi:', ceta,xlag,cphi
649  ! a2:
650  xa2 = (czeta*sin(xlag))/(omega*ctheta)
651  xa2 = xa2/3600 ! convert the unit from s-1 to h-1
652 
653  ! a3:
654  xa3 = msd*(xalb - 1)*(xeta + (ft - fl*xeta)/f*xa1)
655 
656  ! print*, 'sfc_typ_water:', sfc_typ
657  ! print*, 'a1,a2,a3:', xa1,xa2,xa3
658 
659  END SUBROUTINE anohm_coef_water_cal
