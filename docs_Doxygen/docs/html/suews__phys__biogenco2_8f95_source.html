93  ! Calculate PAR from Kdown ----------------------------------

95  vegfracsum = sfr(conifsurf) + sfr(decidsurf) + sfr(grasssurf)

97  ! Calculate active vegetation surface -----------------------
98  !Now this is zero always when LAI in its minimum values. This needs to vary between the summer time maximum and some minimum value
99  !especially in the case of evergreen trees (i.e. in early March LAI can be in its minimum value, but air temperature and radiation
100  !such that uptake can take place)
101  DO iv = ivconif, ivgrass !For vegetated surfaces. Snow included although quite often LAI will be in its minimum when snow on ground
102  active_veg_fr(iv) = (sfr(iv + 2)*(1 - snowfrac(iv + 2)))*(lai_id(iv)/laimax(iv))
103  active_veg_fr0(iv) = (sfr(iv + 2)*(1 - snowfrac(iv + 2)))*lai_id(iv)
104  ENDDO
105 
106  IF (emissionsmethod >= 11 .AND. emissionsmethod <= 16) THEN ! Rectangular hyperbola
107  ! Calculate carbon uptake due to photosynthesis -------------
108  fc_photo = 0
109  DO iv = ivconif, ivgrass
110  fc_photo_surf(iv) = -beta_bioco2(iv)*alpha_bioco2(iv)*par_umolm2s1/(alpha_bioco2(iv)*par_umolm2s1 + beta_bioco2(iv))
111  ! For active vegetation fraction only
112  fc_photo = fc_photo + fc_photo_surf(iv)*active_veg_fr(iv) !umol m-2 s-1
113  ENDDO
114 
115  ELSEIF (emissionsmethod >= 21 .AND. emissionsmethod <= 26) THEN !Local model, Bellucco et al. (2017)
116  ! Calculate carbon uptake due to photosynthesis -------------
117  bellucco2017_pho = 0
118  DO iv = ivconif, ivgrass
119  bellucco2017_pho_surf(iv) = -(1/(2*theta_bioco2(iv)) &
120  *(alpha_bioco2(iv)*par_umolm2s1 + beta_bioco2(iv) &
121  - sqrt((alpha_bioco2(iv)*par_umolm2s1 + beta_bioco2(iv))**2 &
122  - 4*alpha_bioco2(iv)*beta_bioco2(iv)*theta_bioco2(iv)*par_umolm2s1)))
123 
124  ! For active vegetation fraction only
125  bellucco2017_pho = bellucco2017_pho + bellucco2017_pho_surf(iv)*active_veg_fr(iv)
126  ENDDO
127  fc_photo = bellucco2017_pho
128 
129  ELSEIF (emissionsmethod >= 31 .AND. emissionsmethod <= 36) THEN !General model, Bellucco et al. (2017)
130  ! Not currently recommended as includes also some anthropogenic impacts. Should maybe be separate from other biogen choices?
131  ! Alpha and beta calculated as a function of vegetation cover fraction
132 
133  !Different alpha, beta and theta vegetation cover values should be same in BiogenCO2Method = 3.
134  IF (alpha_bioco2(ivconif) == alpha_bioco2(ivdecid) .AND. alpha_bioco2(ivconif) == alpha_bioco2(ivgrass) .AND. &
135  beta_bioco2(ivconif) == beta_bioco2(ivdecid) .AND. beta_bioco2(ivconif) == beta_bioco2(ivgrass) .AND. &
136  theta_bioco2(ivconif) == theta_bioco2(ivdecid) .AND. theta_bioco2(ivconif) == theta_bioco2(ivgrass)) THEN

139  alpha_bioco2_v2(ivconif) = alpha_bioco2(ivconif) + alpha_enh_bioco2(ivconif)* &
140  (sfr(conifsurf) + sfr(decidsurf) + sfr(grasssurf) + sfr(bsoilsurf)) !umol CO2 umol photons-1
141  beta_bioco2_v2(ivconif) = -beta_bioco2(ivconif) + beta_enh_bioco2(ivconif)* &
142  (sfr(conifsurf) + sfr(decidsurf) + sfr(grasssurf) + sfr(bsoilsurf)) !umol m^-2 s^-1
143 
144  !Photosynthesis
145  bellucco2017_pho = -(1/(2*theta_bioco2(ivconif)) &
146  *(alpha_bioco2_v2(ivconif)*par_umolm2s1 &
147  + beta_bioco2_v2(ivconif) &
148  - sqrt((alpha_bioco2_v2(ivconif)*par_umolm2s1 + beta_bioco2_v2(ivconif))**2 - 4* &
149  alpha_bioco2_v2(ivconif)*beta_bioco2_v2(ivconif)*theta_bioco2(ivconif)*par_umolm2s1)))
150 
151  ELSE !If values are not same, then weighted average is calculated.
152  ! CALL ErrorHint(74, 'Check values in SUEWS_BiogenCO2.txt: ', notUsed, notUsed, notUsedI)
153 
154  ! Weighted averages
155  alpha_bioco2_v2(ivconif) = (alpha_bioco2(ivconif)*sfr(conifsurf)/vegfracsum + &
156  alpha_bioco2(ivdecid)*sfr(decidsurf)/vegfracsum &
157  + alpha_bioco2(ivgrass)*sfr(grasssurf)/vegfracsum) &
158  /(alpha_bioco2(ivconif) + alpha_bioco2(ivdecid) + alpha_bioco2(ivgrass))
159  beta_bioco2_v2(ivconif) = (beta_bioco2(ivconif)*sfr(conifsurf)/vegfracsum + &
160  beta_bioco2(ivdecid)*sfr(decidsurf)/vegfracsum &
161  + beta_bioco2(ivgrass)*sfr(grasssurf)/vegfracsum) &
162  /(beta_bioco2(ivconif) + beta_bioco2(ivdecid) + beta_bioco2(ivgrass))
163  theta_bioco2_v2(ivconif) = (theta_bioco2(ivconif)*sfr(conifsurf)/vegfracsum + &

166  /(theta_bioco2(ivconif) + theta_bioco2(ivdecid) + theta_bioco2(ivgrass))

170  (sfr(conifsurf) + sfr(decidsurf) + sfr(grasssurf) + sfr(bsoilsurf)) !umol CO2 umol photons-1

177  - sqrt((alpha_bioco2_v2(ivconif)*par_umolm2s1 + beta_bioco2_v2(ivconif))**2 &

180  ENDIF

183  bellucco2017_pho*active_veg_fr(decidsurf - 2) + &
184  bellucco2017_pho*active_veg_fr(grasssurf - 2)

186  ELSEIF (emissionsmethod >= 41 .AND. emissionsmethod <= 46) THEN ! Conductance parameters

197  ENDIF
198 
199  ENDIF
200

202  bellucco2017_res = 0.0
203  bellucco2017_res_surf = 0.0
204  IF (vegfracsum > 0.01) THEN
205  DO iv = ivconif, ivgrass
206  IF (sfr(2 + iv) > 0.005) THEN
207  IF (gsmodel == 1 .OR. gsmodel == 2) THEN !With air temperature
208  bellucco2017_res_surf = max(min_res_bioco2(iv), resp_a(iv)*exp(resp_b(iv)*temp_c))
209  ELSEIF (gsmodel == 3 .OR. gsmodel == 4) THEN !With modelled 2 meter temperature
210  bellucco2017_res_surf = max(min_res_bioco2(iv), resp_a(iv)*exp(resp_b(iv)*t2))
211  ENDIF
212  ! Only for vegetation fraction
213  bellucco2017_res = bellucco2017_res + bellucco2017_res_surf*sfr(2 + iv)/vegfracsum
214  ENDIF
215  ENDDO
216  ENDIF
217  fc_respi = bellucco2017_res*(sfr(conifsurf) + sfr(decidsurf) + sfr(grasssurf) + sfr(bsoilsurf))
218  ! Combine to find biogenic CO2 flux
219  fc_biogen = fc_photo + fc_respi
220 
221  RETURN
