114  !! These variables don't seem to be needed (commented out HCW 27 Nov 2014)

121  ! Daily min and max temp (these get updated through the day) ---------------------
122  gdd(id,3) = min(temp_c,gdd(id,3)) !Daily min T in column 3
123  gdd(id,4) = max(temp_c,gdd(id,4)) !Daily max T in column 4
124  IF (avkdn>10) THEN
125  gdd(id,5) = gdd(id,5)+1/nsh_real !Cumulate daytime hours !Divide by nsh (HCW 01 Dec 2014)
126  ENDIF

128  ! Calculations related to heating and cooling degree days (HDD) ------------------
129  ! See Sailor & Vasireddy (2006) EMS Eq 1,2 (theirs is hourly timestep)
130  IF ((basethdd-temp_c)>=0) THEN !Heating
131  gamma1=1
132  ELSE
133  gamma1=0
134  ENDIF
135 
136  IF ((temp_c-basethdd)>=0) THEN !Cooling
137  gamma2=1
138  ELSE
139  gamma2=0
140  ENDIF
141 
142  IF(gridiv == 1) tstepcount=tstepcount+1 !Add 1 to tstepcount only once for all grids
143 
144  hdd(id,1)=hdd(id,1) + gamma1*(basethdd-temp_c) !Heating
145  hdd(id,2)=hdd(id,2) + gamma2*(temp_c-basethdd) !Cooling
146  hdd(id,3)=hdd(id,3) + temp_c !Will become daily average temperature
147  ! 4 ------------------------------------! !5-day running mean
148  hdd(id,5)=hdd(id,5) + precip !Daily precip total
149  ! 6 ------------------------------------! !Days since rain
150 
151  ! Update snow density, albedo surface fraction
152  IF (snowuse==1) CALL snowupdate(temp_c)
153 
154 
155  ! ================================================================================
156  ! This next part occurs only on the first or last timestep of each day ===========
157 
158  ! --------------------------------------------------------------------------------
159  ! On first timestep of each day, define whether the day each a workday or weekend
160  IF (it==0.AND.imin==0) THEN
161 
162  CALL day2month(id,mb,date,seas,iy,lat) !Calculate real date from doy
163  CALL day_of_week(date,mb,iy,wd) !Calculate weekday (1=Sun, ..., 7=Sat)
164 
165  dayofweek(id,1)=wd !Day of week
166  dayofweek(id,2)=mb !Month
167  dayofweek(id,3)=seas !Season
168 
169  ! --------------------------------------------------------------------------------
170  ! On last timestep, perform the daily calculations -------------------------------
171  ! Daily values not correct until end of each day,
172  ! so main program should use values from the previous day
173  ELSEIF (it==23 .AND. imin==(nsh_real-1)/nsh_real*60) THEN
174  !write(*,*) 'Last timestep of day'
175 
176  ! Heating degree days (HDD) -------------
177  hdd(id,1)=hdd(id,1)/tstepcount !Heating
178  hdd(id,2)=hdd(id,2)/tstepcount !Cooling
179  hdd(id,3)=hdd(id,3)/tstepcount !Average temp
180 
181  IF(gridiv == numberofgrids) tstepcount=0 !Set to zero only after last grid has run
182 
183  ! Calculate 5-day running mean temp !!Need to deal with the previous year - CHECK!!
184  DO jj=1,5
185  hdd(id,4)=hdd(id,4) + hdd(id-(jj-1),3)
186  ENDDO
187  hdd(id,4) = hdd(id,4)/5
188 
189  ! Calculate number of days since rain
190  IF(hdd(id,5)>0) THEN !Rain occurred
191  hdd(id,6)=0
192  ELSE
193  hdd(id,6)=hdd(id-1,6)+1 !Days since rain
194  ENDIF
195 
196 
197  ! Calculate modelled daily water use ------------------------------------------
198  IF (waterusemethod==0) THEN !If water use is to be modelled (rather than observed)
199 
200  wd=dayofweek(id,1)
201 
202  IF (daywat(wd)==1.0) THEN !1 indicates watering permitted on this day
203  calc=0
204  IF (lat>=0) THEN !Northern Hemisphere
205  IF (id>=ie_start-1.AND.id<=ie_end+1) calc=1 !Day between irrigation period
206  ELSE !Southern Hemisphere
207  calc=1
208  IF (id>=ie_end.AND.id<=ie_start) calc=0 !Day between irrigation period
209  ENDIF
210 
211  IF(calc==1) THEN
212  ! Model daily water use based on HDD(id,6)(days since rain) and HDD(id,3)(average temp)
213  ! WU_Day is the amount of water [mm] per day, applied to each of the irrigated areas
214  ! N.B. These are the same for each vegetation type at the moment
215 
216  ! ---- Automatic irrigation (evergreen trees) ----
217  wu_day(id,2) = faut*(ie_a(1)+ie_a(2)*hdd(id,3)+ie_a(3)*hdd(id,6))*daywatper(wd)
218  IF (wu_day(id,2)<0) wu_day(id,2)=0 !If modelled WU is negative -> 0
219 
220  ! ---- Manual irrigation (evergreen trees) ----
221  wu_day(id,3) = (1-faut)*(ie_m(1)+ie_m(2)*hdd(id,3)+ie_m(3)*hdd(id,6))*daywatper(wd)
222  IF (wu_day(id,3)<0) wu_day(id,3)=0 !If modelled WU is negative -> 0
223 
224  ! ---- Total evergreen trees water use (automatic + manual) ----
225  wu_day(id,1)=(wu_day(id,2)+wu_day(id,3))
226 
227  ! ---- Automatic irrigation (deciduous trees) ----
228  wu_day(id,5) = faut*(ie_a(1)+ie_a(2)*hdd(id,3)+ie_a(3)*hdd(id,6))*daywatper(wd)
229  IF (wu_day(id,5)<0) wu_day(id,5)=0 !If modelled WU is negative -> 0
230 
231  ! ---- Manual irrigation (deciduous trees) ----
232  wu_day(id,6) = (1-faut)*(ie_m(1)+ie_m(2)*hdd(id,3)+ie_m(3)*hdd(id,6))*daywatper(wd)
233  IF (wu_day(id,6)<0) wu_day(id,6)=0 !If modelled WU is negative -> 0
234 
235  ! ---- Total deciduous trees water use (automatic + manual) ----
236  wu_day(id,4)=(wu_day(id,5)+wu_day(id,6))
237 
238  ! ---- Automatic irrigation (grass) ----
239  wu_day(id,8) = faut*(ie_a(1)+ie_a(2)*hdd(id,3)+ie_a(3)*hdd(id,6))*daywatper(wd)
240  IF (wu_day(id,8)<0) wu_day(id,8)=0 !If modelled WU is negative -> 0
241 
242  ! ---- Manual irrigation (grass) ----
243  wu_day(id,9) = (1-faut)*(ie_m(1)+ie_m(2)*hdd(id,3)+ie_m(3)*hdd(id,6))*daywatper(wd)
244  IF (wu_day(id,9)<0) wu_day(id,9)=0 !If modelled WU is negative -> 0
245 
246  ! ---- Total grass water use (automatic + manual) ----
247  wu_day(id,7)=(wu_day(id,8)+wu_day(id,9))
248 
249  ELSE !If no irrigation on this day
250  wu_day(id,1)=0
251  wu_day(id,2)=0
252  wu_day(id,3)=0
253  wu_day(id,4)=0
254  wu_day(id,5)=0
255  wu_day(id,6)=0
256  wu_day(id,7)=0
257  wu_day(id,8)=0
258  wu_day(id,9)=0
259  ENDIF
260  ENDIF
261  ENDIF
262 
263  !------------------------------------------------------------------------------
264  ! Calculation of LAI from growing degree days
265  ! This was revised and checked on 16 Feb 2014 by LJ
266  !------------------------------------------------------------------------------
267 
268  critdays=50 !Critical limit for GDD when GDD or SDD is set to zero
269 
270  ! Loop through vegetation types (iv)
271  DO iv=1,nvegsurf
272  ! Calculate GDD for each day from the minimum and maximum air temperature
273  yes =((gdd(id,3)+gdd(id,4))/2-baset(iv)) !Leaf on
274  no =((gdd(id,3)+gdd(id,4))/2-basete(iv)) !Leaf off
275 
276  indhelp = 0 !Help switch to allow GDD to go to zero in sprint-time !!What does this mean?? HCW
277 
278  IF(yes<0) THEN !GDD cannot be negative
279  indhelp=yes !Amount of negative GDD
280  yes=0
281  ENDIF
282 
283  IF(no>0) no=0 !SDD cannot be positive
284 
285  ! Calculate cumulative growing and senescence degree days
286  gdd(id,1) = gdd(id-1,1)+yes
287  gdd(id,2) = gdd(id-1,2)+no
288 
289  ! Possibility for cold spring
290  IF(gdd(id,2)<=sddfull(iv).AND.indhelp<0) THEN
291  gdd(id,1)=0
292  ENDIF
293 
294  IF(gdd(id,1)>=gddfull(iv)) THEN !Start senescence
295  gdd(id,1)=gddfull(iv) !Leaves should not grow so delete yes from earlier
296  IF(gdd(id,2)<-critdays) gdd(id,1)=0
297  ENDIF
298 
299  IF (gdd(id,2)<=sddfull(iv)) THEN !After senescence now start growing leaves
300  gdd(id,2)=sddfull(iv) !Leaves off so add back earlier
301  IF(gdd(id,1)>critdays) gdd(id,2)=0
302  ENDIF
303 
304  ! With these limits SDD, GDD is set to zero
305  IF(gdd(id,2)<-critdays.AND.gdd(id,2)>sddfull(iv)) gdd(id,1)=0
306  IF(gdd(id,1)> critdays.AND.gdd(id,1)<gddfull(iv)) gdd(id,2)=0
307 
308  ! Now calculate LAI itself
309  IF(lat>=0) THEN !Northern hemispere
310  IF (id==140.AND.gdd(id,2)/=0) gdd(id,2)=0 !If SDD is not zero by mid May, this is forced
311  ! Set SDD to zero in summer time
312  IF (gdd(id,1)> critdays.AND.id<170) gdd(id,2)=0
313  ! Set GDD zero in winter time
314  IF (gdd(id,2)<-critdays.AND.id>170) gdd(id,1)=0
315 
316  IF (laitype(iv) < 0.5) THEN !Original LAI type
317  IF(gdd(id,1)>0.AND.gdd(id,1)<gddfull(iv)) THEN !Leaves can still grow
318  lai(id,iv)=(lai(id-1,iv)**laipower(1,iv)*gdd(id,1)*laipower(2,iv))+lai(id-1,iv)
319  ELSEIF(gdd(id,2)<0.AND.gdd(id,2)>sddfull(iv)) THEN !Start senescence
320  lai(id,iv)=(lai(id-1,iv)**laipower(3,iv)*gdd(id,2)*laipower(4,iv))+lai(id-1,iv)
321  ELSE
322  lai(id,iv)=lai(id-1,iv)
323  ENDIF
324  ELSEIF (laitype(iv)>=0.5) THEN
325  IF(gdd(id,1)>0.AND.gdd(id,1)<gddfull(iv)) THEN !Leaves can still grow
326  lai(id,iv)=(lai(id-1,iv)**laipower(1,iv)*gdd(id,1)*laipower(2,iv))+lai(id-1,iv)
327  !! Use day length to start senescence at high latitudes (N hemisphere)
328  ELSEIF (gdd(id,5)<=12.AND.gdd(id,2)>sddfull(iv)) THEN !Start senescence
329  lai(id,iv)=(lai(id-1,iv)*laipower(3,iv)*(1-gdd(id,2))*laipower(4,iv))+lai(id-1,iv)
330  ELSE
331  lai(id,iv)=lai(id-1,iv)
332  ENDIF
333  ENDIF
334 
335  ELSEIF (lat<0) THEN !Southern hemisphere !! N.B. not identical to N hemisphere - return to later
336  IF (id==300.AND.gdd(id,2)/=0) gdd(id,2)=0 !If SDD is not zero by late Oct, this is forced
337  ! Set SDD to zero in summer time
338  IF (gdd(id,1)> critdays.AND.id>250) gdd(id,2)=0
339  ! Set GDD zero in winter time
340  IF (gdd(id,2)<-critdays.AND.id<250) gdd(id,1)=0
341 
342  IF (laitype(iv) < 0.5) THEN !Original LAI type
343  IF(gdd(id,1)>0.AND.gdd(id,1)<gddfull(iv)) THEN
344  lai(id,iv)=(lai(id-1,iv)**laipower(1,iv)*gdd(id,1)*laipower(2,iv))+lai(id-1,iv)
345  ELSEIF(gdd(id,2)<0.AND.gdd(id,2)>sddfull(iv)) THEN
346  lai(id,iv)=(lai(id-1,iv)**laipower(3,iv)*gdd(id,2)*laipower(4,iv))+lai(id-1,iv)
347  ELSE
348  lai(id,iv)=lai(id-1,iv)
349  ENDIF
350  ELSE
351  IF(gdd(id,1)>0.AND.gdd(id,1)<gddfull(iv)) THEN
352  lai(id,iv)=(lai(id-1,iv)**laipower(1,iv)*gdd(id,1)*laipower(2,iv))+lai(id-1,iv)
353  !! Day length not used to start senescence in S hemisphere (not much land)
354  ELSEIF(gdd(id,2)<0.AND.gdd(id,2)>sddfull(iv)) THEN
355  lai(id,iv)=(lai(id-1,iv)*laipower(3,iv)*(1-gdd(id,2))*laipower(4,iv))+lai(id-1,iv)
356  ELSE
357  lai(id,iv)=lai(id-1,iv)
358  ENDIF
359  ENDIF
360  ENDIF !N or S hemisphere
361 
362  ! Check LAI within limits; if not set to limiting value
363  IF(lai(id,iv).GT.laimax(iv))THEN
364  lai(id,iv)=laimax(iv)
365  ELSEIF(lai(id,iv).LT.laimin(iv))THEN
366  lai(id,iv)=laimin(iv)
367  ENDIF
368 
369  ENDDO !End of loop over veg surfaces
370  !------------------------------------------------------------------------------
371 
372  !::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
373  ! Calculate the development of vegetation cover
374  ! Albedo changes with LAI for each vegetation type
375  ! Storage capacity and porosity are updated based on DecTr LAI only (seasonal variation in Grass and EveTr assumed small)
376  ! If only LUMPS is used, set deciduous capacities to 0
377  ! Assume porosity Change based on GO99- Heisler??
378  deltalai=0
379  deltalaievetr=0
380  deltalaigrass=0
381  capchange=0
382  porchange=0
383  albchangedectr=0
384  albchangeevetr=0
385  albchangegrass=0
386 
387  iv=ivdecid
388  IF((lai(id,iv)-lai(id-1,iv))/=0) THEN
389  deltalai=(lai(id,iv)-lai(id-1,iv))/(laimax(iv)-laimin(iv))
390  albchangedectr=(albmax_dectr-albmin_dectr)* deltalai
391  capchange=(capmin_dec-capmax_dec)* deltalai
392  porchange=(pormin_dec-pormax_dec)* deltalai
393  ENDIF
394 
395  iv=ivconif
396  IF((lai(id,iv)-lai(id-1,iv))/=0) THEN
397  deltalaievetr=(lai(id,iv)-lai(id-1,iv))/(laimax(iv)-laimin(iv))
398  albchangeevetr=(albmax_evetr-albmin_evetr)* deltalaievetr !!N.B. Currently uses deltaLAI for deciduous trees only!!
399  ENDIF
400 
401  iv=ivgrass
402  IF((lai(id,iv)-lai(id-1,iv))/=0) THEN
403  deltalaigrass=(lai(id,iv)-lai(id-1,iv))/(laimax(iv)-laimin(iv))
404  albchangegrass=(albmax_grass-albmin_grass)* deltalaigrass !!N.B. Currently uses deltaLAI for deciduous trees only!!
405  ENDIF
406 
407  iv=ivdecid
408 
409  !write(*,*) deltaLAI, deltaLAIEveTr, deltaLAIGrass
410 
411  decidcap(id) = decidcap(id-1) - capchange
412  albdectr(id) = albdectr(id-1) + albchangedectr
413  porosity(id) = porosity(id-1) + porchange !- changed to + by HCW 20 Aug 2015 (porosity greatest when LAI smallest)
414  !Also update albedo of EveTr and Grass surfaces
415  albevetr(id) = albevetr(id-1) + albchangeevetr
416  albgrass(id) = albgrass(id-1) + albchangegrass
417 
418  ! ------- AnOHM related, added by TS 01 Mar 2016 ------------------------------
419  ! read in QH and QE of the current day
420  ALLOCATE(lsub(SIZE(dataout(:,2,gridiv))))
421  WHERE (dataout(:,2,gridiv) == id)
422  lsub = 1
423  ELSEWHERE
424  lsub = 0
425  END WHERE
426  irrange = (/maxloc(lsub),maxloc(lsub)+sum(lsub)/)
427  ! write(*,*) 'good here: 424',irRange(2), irRange(1),sum(lSub),maxloc(lSub)
428  ! calculate Bowen ratio
429  ! check if the diurnal cycle of present day (id) is complete
430  IF (irrange(2)- irrange(1) >= nsh*24-1) THEN
431  ! load the sublist into forcings:
432  ! QH and QE for Bowen ratio
433 
434  ! needs improvement:
435  ! TODO: include all timesteps as in the calculations
436  xqh = dataout(irrange(1):irrange(2):nsh,14,gridiv)
437  xqe = dataout(irrange(1):irrange(2):nsh,15,gridiv)
438  mxqh = sum(xqh(10:16))/7
439  mxqe = sum(xqe(10:16))/7
440 
441 
442  ! handle extreme dry condition to prevent NAN
443  IF ( abs(mxqe) < 0.1 ) mxqe = 0.1
444 
445  xbo = mxqh/mxqe
446  ! calculate daily mean AH
447  xah = dataout(irrange(1):irrange(2):nsh,15,gridiv)
448  xmah = sum(xah(:))/24
449  ELSE
450  ! give a default Bo as 2 if no enough flux data to update with:
451  xbo = 10.
452  xmah = 25.
453  END IF
454  ! handle NAN
455  IF ( xbo/=xbo ) THEN
456  xbo = 2.
457  END IF
458  IF ( xmah/=xmah ) THEN
459  xmah = 25.
460  END IF
461 
462 
463  boanohmend(gridiv) = xbo
464  bo_grids(id,gridiv) = boanohmend(gridiv)
465  boinit = bo_grids(id,gridiv) ! BoInit will be written out to InitialConditions of next year
