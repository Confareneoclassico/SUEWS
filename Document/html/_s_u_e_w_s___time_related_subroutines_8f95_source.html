216 ! Calculation of declination of sun (Eqn. 16). Amplitude= +/-23.45
217 ! deg. Minimum = DOY 355 (DEC 21), maximum = DOY 172.5 (JUN 21/22).
218  dec = -23.45 * cos(2.0*pi*(doy+10.0)/365.0)
219 
220 ! Sun angles. SOC limited for latitudes above polar circles.
221  soc = tan(rad*dec) * tan(rad*xlat)
222  soc = min(max(soc,-1.0),1.0)

224 ! Calculate daylength, sunrise and sunset (Eqn. 17)
