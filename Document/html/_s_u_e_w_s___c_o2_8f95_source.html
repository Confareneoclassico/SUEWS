2  SUBROUTINE co2_biogen
3 ! Created by HCW Aug 2016 to calculate biogenic component of CO2 flux.
4 ! -ve Fc for uptake; +ve Fc for emission
5 !
6 ! Last modified:
7 ! HCW 11 Apr 2017 - Tidied and merged with LJ code
8 ! LJ 6 Apr 2017 - Minimum limit for soil respiration with BiogenCO2Choice = 2 was set to 0.6 umol m-2 s-1
9 ! - Choice for non-rectancular hyperbola to calculate the biogenic CO2 flux added (BiogenCO2Choice = 2)
10 ! (Bellucco et al. 2017, Agric. Forest. Met. 236, 113-122).
11 ! - Both the "local Helsinki model" (BiogenCO2Choice = 2)) and "general model" BiogenCO2Choice = 3) are implemented
12 ! - Snow fraction added to the calculation of active vegetation fraction and the soil respiration
13 !
14 ! To Do:
15 ! - Active vegetation goes to zero with LAI minimum, but this needs to be changed so some minimum value
16 ! especially in the case of evergreentrees
17 ! - Move some of the parameters to input files
18 ! - Now on weekend nighttime population is used throughout the day. Do we need extra column in SiteSelect for daytime weekend population?
19 !========================================================================================
20 
21  USE allocatearray
22  USE data_in
23  USE physconstants
24  USE time
25 
26  IMPLICIT NONE
27 
28  INTEGER:: iv ! counter
29  INTEGER:: BiogenCO2Choice = 1 !Move to RunControl later
30  ! 1 - Rectangular hyperbola (Ruimy, Schmid, Flanagan)
31  ! 2 - Non-rectangular hyperbola, Helsinki (Bellucco et al. 2016)
32  ! 3 - Non-rectangular hyperbola, general (Bellucco et al. 2016)
33 
34  REAL(KIND(1d0)):: PAR_umolm2s1
35 
36  REAL(KIND(1d0)):: KdnToPAR ! Conversion from Kdn to PAR
37 
38  REAL(KIND(1d0)):: Min_respi ! Minimum soil respiration rate (for cold-temperature limit)
39 
40  ! Coefficients for rectangular hyperbola light response curve (for each surface)
41  REAL(KIND(1d0)),DIMENSION(nvegsurf):: RecHyp_AMax ! umol CO2 m-2 s-1
42  REAL(KIND(1d0)),DIMENSION(nvegsurf):: RecHyp_alpha ! umol CO2 (umol photons)-1
43 
44  ! Coefficients for non-rectangular hyperbola (Bellucco et al. 2016)
45  REAL(KIND(1d0)):: alpha_NRH, beta_NRH, theta_NRH
46 
47  REAL(KIND(1d0)):: Bellucco2016_Pho, Bellucco2016_Res
48 
49  REAL(KIND(1d0)),DIMENSION(nvegsurf):: active_veg_fr
50  REAL(KIND(1d0)),DIMENSION(nvegsurf):: Fc_photo_surf ! Photosynthesis for each vegetated surface
51 
52  ! Define coefficients (could be moved elsewhere) ------------
53  !!! Tidy these up (nomenclature and signs)
54  !KdnToPAR = 0.473 !Papaioannou et al. (1993) (mean annual value)
55  kdntopar = 0.46 !From Tsubo and Walker, 2005: PAR is on average 0.46 Kdown
56 
57  min_respi = 0.6 ! LJ & UHEL (0.6 umol m-2 s-1 estimated from Hyytiala forest site)
58 
59  ! Calculate PAR from Kdown ----------------------------------
60  par_umolm2s1 = jtoumolpar * kdntopar * avkdn
61 
62  ! Calculate active vegetation surface -----------------------
63  !Now this is zero always when LAI in its minimum values. This needs to vary between the summer time maximum and some minimum value
64  !especially in the case of evergreen trees (i.e. in early March LAI can be in its minimum value, but air temperature and radiation
65  !such that uptake can take place)
66  DO iv=ivconif,ivgrass !For vegetated surfaces. Snow included although quite often LAI will be in its minimum when snow on ground
67  active_veg_fr(iv) = (sfr(iv+2)*(1-snowfrac(iv+2)))*(lai(id-1,iv)-laimin(iv))/(laimax(iv)-laimin(iv))
68  ENDDO
69 
70 
71  IF(biogenco2choice == 1) THEN ! Rectangular hyperbola
72  ! Set these in input files
73  !! Schmid et al. (2000) empirical model for hardwood forest (Eq 6. with dark respiration rate removed)
74  !RecHyp_AMax(:) = 35
75  !RecHyp_alpha(:) = 0.0593
76  !! Flanagan et al. (2002) empirical model for temperate grassland (mean of summer 1998, 1999, 2000)
77  !RecHyp_AMax(:) = 16.3
78  !RecHyp_alpha(:) = 0.0205
79  ! Ruimy et al. (1995) review for plant canopies
80  rechyp_amax(:) = 43.35
81  rechyp_alpha(:) = 0.044
82 
83  ! Calculate carbon uptake due to photosynthesis -------------
84  fc_photo = 0
85  DO iv=ivconif,ivgrass
86  fc_photo_surf(iv) = -rechyp_amax(iv)*rechyp_alpha(iv)*par_umolm2s1/(rechyp_alpha(iv)*par_umolm2s1 + rechyp_amax(iv))
87  ! For active vegetation fraction only
88  fc_photo = fc_photo + fc_photo_surf(iv)*active_veg_fr(iv) !umol m-2 s-1
89  ENDDO
90 
91  ! Calculate ecosystem respiration ---------------------------
92  ! Eq 5 Schmid et al. (2000), with minimum value from UHEL
93  fc_respi = max(min_respi, 1.08*exp(0.064*temp_c)) !umol m-2 s-1 !!!Switch to using soil temp?
94  ! For natural surfaces only (with no snow)
95  fc_respi = fc_respi* (sfr(conifsurf)*(1-snowfrac(conifsurf))+sfr(decidsurf)*(1-snowfrac(decidsurf))+ &
96  sfr(grasssurf)*(1-snowfrac(grasssurf))+sfr(bsoilsurf)*(1-snowfrac(bsoilsurf)))
97 
98  ELSEIF(biogenco2choice == 2 .OR. biogenco2choice == 3) THEN !Bellucco et al. (2016)
99  IF(biogenco2choice == 2) THEN ! Local Helsinki model
100  alpha_nrh = 0.031 ! umol CO2 umol photons-1
101  beta_nrh = 17.793 ! umol m^-2 s^-1
102  theta_nrh = 0.723 ! -
103 
104  ! Respiration calculated from Helsinki (Järvi et al. 2012), 0.6 limit estimated from Hyytiala forest site
105  bellucco2016_res = max(min_respi, 3.229*exp(0.0329*temp_c))
106 
107  ELSEIF (biogenco2choice == 3) THEN ! General model
108  ! Not currently recommended as includes also some anthropogenic impacts. Should maybe be separate from other biogen choices?
109 
110  ! Alpha and beta calculated as a function of vegetation cover fraction
111  alpha_nrh = 0.005 + 0.016*(sfr(conifsurf)+sfr(decidsurf)+sfr(grasssurf)+sfr(bsoilsurf)) !umol CO2 umol photons-1
112  beta_nrh = -8.474 + 33.454*(sfr(conifsurf)+sfr(decidsurf)+sfr(grasssurf)+sfr(bsoilsurf)) !umol m^-2 s^-1
113 
114  theta_nrh = 0.96 ! -
115 
116  ! Respiration set as fixed value
117  bellucco2016_res = 2.43 ! umol m^-2 s^-1
118 
119  ENDIF
120 
121  ! Photosynthesis calculated using NRH
122  bellucco2016_pho = -(1/(2*theta_nrh)*(alpha_nrh*par_umolm2s1+beta_nrh- &
123  sqrt((alpha_nrh*par_umolm2s1+beta_nrh)**2-4*alpha_nrh*beta_nrh*theta_nrh*par_umolm2s1)))
124 
125  fc_photo = bellucco2016_pho*active_veg_fr(conifsurf-2)+ &
126  bellucco2016_pho*active_veg_fr(decidsurf-2)+ &
127  bellucco2016_pho*active_veg_fr(grasssurf-2)
128 
129  fc_respi = bellucco2016_res * (sfr(conifsurf)*(1-snowfrac(conifsurf))+sfr(decidsurf)*(1-snowfrac(decidsurf))+ &
130  sfr(grasssurf)*(1-snowfrac(grasssurf))+sfr(bsoilsurf)*(1-snowfrac(bsoilsurf)))
131 
132  ENDIF
133 
134  ! Combine to find biogenic CO2 flux
135  fc_biogen = fc_photo + fc_respi
136 
137  RETURN
138 
139  ENDSUBROUTINE co2_biogen

144  SUBROUTINE co2_anthro(id, ih, imin)
145 ! Created by HCW Aug 2016 to calculate anthropogenic component of CO2 flux.
146 ! -ve Fc for uptake; +ve Fc for emission
147 !
148 ! To Do:
149 ! - Add specific diurnal profiles for traffic and building energy use for AnthropCO2Method=3
150 ! (currently the same anthropogenic heat profile is applied to both traffic and buildings)
151 ! - Add anthropogenic latent heat too?
152 !========================================================================================
153 
154  USE allocatearray
155  USE data_in
156  USE defaultnotused
157  USE sues_data
158 
159  IMPLICIT NONE
160 
161  INTEGER:: iu ! 1=weekday or 2=weekend
162  INTEGER:: ih ! hour accounting for DLS
163  INTEGER:: id, imin
164 
165  REAL(KIND(1d0)):: ActDorNorT, PopDorNorT !Daytime, night-time or transition time
166 
167  REAL(KIND(1d0)):: QF_build, QF_traff, QF_metab
168 
169  !!!Move to input file later!!! Rename SUEWS_AnthropogenicHeat.txt to SUEWS_Anthropogenic.txt and add these there?
170  REAL(KIND(1d0)):: FcEF_v_kgkm, EnEF_v_Jkm, EF_umolCO2perJ, FracFossilFuel, QF0_NB
171 
172  !AnthropCO2Method
173  !2 - CO2 emissions based on QF calculated according to AnthropHeatMethod=2;
174  !3 - CO2 emissions based on mean traffic rate and building energy use specified in SiteSelect
175 
176  ! Define coefficients --------------------------------------- ! Move to inputs?
177  ! CO2 emission factors
178  fcef_v_kgkm = 0.2069 !Average car, UK (DECC) [kg km-1]
179  !FcEF_v_kgkm = 0.295 !Moriwaki & Kanda (2004) [kg km-1]
180  ! Energy emission factors
181  enef_v_jkm = 3.97e6 ! [J km−1] Sailor & Lu (2004)
182  ! Emission factors for fuels
183  ef_umolco2perj = 51.0/44 !Natural gas (Moriwaki & Kanda 2004)
184  !EF_umolCO2perJ = 67.6/44 !Kerosene (Moriwaki & Kanda 2004)
185 
186  fracfossilfuel = 0.70 !Proportion of building energy use from fossil fuels rather than electricity (0.6-0.8 for Sw)
187  !Some variation with season
188  qf0_nb = 0.25
189 
190  ! Establish whether weekday or weekend ----------------------
191  iu=1 !Set to 1=weekday
192  IF(dayofweek(id,1)==1.or.dayofweek(id,1)==7) THEN
193  iu=2 !Set to 2=weekend
194  ENDIF
195 
196  ! Calculate CO2 emissions from human metabolism -------------
197  ! (Pop densities in ha-1 -> m-2)
198  popdornort = humactivity_tstep((nsh*(ih+1-1)+imin*nsh/60+1),iu) !!! Set separately later !!!
199  IF (iu==2) popdornort = 1 !!!
200  actdornort = humactivity_tstep((nsh*(ih+1-1)+imin*nsh/60+1),iu) !1=night, 2=day, 1-2=transition
201  fc_metab = (popdensnighttime*(2-popdornort) + popdensdaytime*(popdornort-1))/10000 * (120*(2-actdornort) + 280*(actdornort-1)) !umol m-2 s-1
202  qf_metab = (popdensnighttime*(2-popdornort) + popdensdaytime*(popdornort-1))/10000 * (75*(2-actdornort) + 175*(actdornort-1)) !W m-2
203 
204  ! Calculate CO2 emissions from traffic ----------------------
205  IF(anthropco2method == 2) THEN
206  ! Assume fraction QF0_NB of temperature independent part of QF is traffic + metabolism
207  qf_traff = qf0_nb*qf_sahp_base - qf_metab ! need to calculate QF_metab here again!!!
208  IF(qf_traff < 0) THEN
209  CALL errorhint(69,'QF metab exceeds base QF in Fc_anthro.',qf_metab,qf_sahp_base,notusedi)
210  qf_traff = 0
211  ENDIF
212  fc_traff = qf_traff / enef_v_jkm * fcef_v_kgkm*1e3*1e6/44 !Divide QF by energy emission factor and multiply by CO2 factor
213  ELSEIF(anthropco2method == 3) THEN
214  ! Calculate using mean traffic rate [veh km cap-1 day-1] * emission factor [kg km-1] * 1e3 g kg-1 /44 g mol-1 * 1e6 umol mol-1
215  ! Which popdens? !!!
216  fc_traff = popdensnighttime/10000 * trafficrate/(60*60*24) * fcef_v_kgkm*1e3*1e6/44 * &
217  ahprof_tstep((nsh*(ih+1-1)+imin*nsh/60+1),iu)
218  ENDIF
219 
220  ! Calculate CO2 emissions from building energy use ----------
221  IF(anthropco2method == 2) THEN
222  ! Assume temperature independent part of QF is traffic + metabolism, !!! need QB_0_NB here!!!
223  ! CDD part is electric A/C (no local CO2 emissions)
224  ! HDD part is building energy use, split between electric (no local emissions CO2) and combustion (CO2) heating
225  fc_build = qf_sahp_heat * ef_umolco2perj * fracfossilfuel
226  ELSEIF(anthropco2method == 3) THEN
227  ! Calculate using building energy use [W cap-1]
228  fc_build = popdensnighttime/10000 * buildenergyuse * ef_umolco2perj * fracfossilfuel * &
229  ahprof_tstep((nsh*(ih+1-1)+imin*nsh/60+1),iu) !Need to work out how to incorporate temp dependence !!!
230  !Fc_build = Fc_build * 0.7 !Allen et al. (2010)
231  !! Use daily mean temperature from previous day
232  !IF(HDD(id-1,3) < 12.0) THEN
233  ! Fc_build = Fc_build * (1.0 + min(12.0-HDD(id-1,3),12.0--4)*0.05) ! 5% increase in e (N.B. this is for heating only in Allen et al. 2010)!!!
234  ! write(*,*) (1.0 + min(12.0-HDD(id-1,3),12.0--4)*0.05)
235  ! ! Do not apply increase at high temp as this is AC (assume electric) N.B. slightly contradictory with FracFossilFuel
236  !!ELSEIF(HDD(id-1,3) > 12.0) THEN
237  !! Fc_build = Fc_build * (1.0 + min(HDD(id-1,3)-12.0,35-12.0)*0.03) ! 3% increase in e (N.B. this is for AC only in Allen et al. 2010)!!!
238  !ENDIF
239 
240  ! Pigeon et al. (2007) (approx coeffs)
241  IF(hdd(id-1,3) < 15.0) fc_build = fc_build * (1.0 + (15.0-hdd(id-1,3))*0.5)
242  !write(*,*) (1.0 + (15.0-HDD(id-1,3))*0.5)
243 
244  ENDIF
245 
246  ! Sum components to give anthropogenic CO2 flux
247  fc_anthro = fc_metab + fc_traff + fc_build
248 
249  !write(*,*) Fc_anthro, Fc_metab, Fc_traff, Fc_build
250  !write(*,*) QF_SAHP_heat+QF_SAHP_base, QF_metab, QF_traff, QF_SAHP_heat
251 
252  RETURN
253
