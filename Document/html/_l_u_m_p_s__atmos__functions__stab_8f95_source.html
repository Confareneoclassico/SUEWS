131 FUNCTION stab_fn_mom(StabilityMethod,ZL,zl_f) RESULT(psym)
132  ! StabilityMethod = 1-4 -
133  ! PSYM - stability FUNCTION for momentum
134  !Modified by LJ Mar 2010
135  !Input:Used stability method, stability (z-d)/L, zeta (either (z-d)/L or z0/L)
136 
137  ! USE mod_z
138  ! USE mod_k
139 
140  IMPLICIT NONE
141  REAL(KIND(1d0)),PARAMETER :: &
142  k=0.4,& !Von Karman's contant
143  k2=0.16,& !Power of Van Karman's contant
144  neut_limit=0.001000,& !Limit for neutral stability
145  notusedi=-55
146 
147  REAL (KIND(1d0)):: piover2,psym,zl,zl_f,x,x2
148  INTEGER ::StabilityMethod
149 
150  piover2=acos(-1.)/2.
151  !PRINT*,StabilityMethod,zl,"stab_fn_mom:"
152  IF(abs(zl)<neut_limit) THEN
153  psym=0
154  ELSEIF(zl<-neut_limit) THEN !Unstable
155 
156  IF(stabilitymethod==1)THEN ! Jensen et al 1984 - Van Ulden & Holtslag (1985) p 1206&
157  psym=((1.-16.*zl_f)**0.25)-1
158  ELSEIF(stabilitymethod==2) THEN !Dyer (1974)(1-16z/L)**.25' k=0.41 mod. Hogstrom (1988)v15.2
159  x=(1.-(15.2*zl_f))**0.25
160  x2=log((1+(x**2.))/2.)
161  psym=(2.*log((1+x)/2.))+x2-(2.*atan(x))+piover2
162  ELSEIF(stabilitymethod==3)THEN ! campbell & norman eqn 7.26
163  psym=0.6*(2)*log((1+(1-16*zl_f)**0.5)/2)
164  ELSEIF(stabilitymethod==4) THEN !Businger et al (1971) modifed Hogstrom (1988)
165  x=(1-19.3*zl_f)**(-0.25)
166  x2=log((1+(x**2.))/2.)
167  psym=(2.*log((1+x)/2.))+x2-(2.*atan(x))+piover2
168  ELSEIF(stabilitymethod==7) THEN ! Dyer & Bradley (1982) (1-28z/L)**.25' k=0.4
169  x=(1+(28.*zl_f))**0.25
170  x2=log((1+x**2.)/2.)
171  psym=(2.*log((1+x)/2.))+x2-(2.*atan(x))+piover2
172  ELSEIF(stabilitymethod==5)THEN ! Zilitinkevich & Chalikov (1968) modified Hogstrom (1988)
173  IF(zl_f>=-0.16)THEN
174  x=1+1.38*zl_f
175  ELSE
176  x=0.42*(-1)*zl_f**0.333
177  ENDIF
178  x2=log((1+(x**2.))/2.)
179  psym=(2.*log((1+x)/2.))+x2-(2.*atan(x))+piover2
180 
181  ELSEIF(stabilitymethod==6)THEN ! Foken and Skeib (1983)
182  IF(zl_f>=0.06)THEN
183  x=1
184  ELSE
185  x=((-1)*zl_f/0.06)**0.25
186  ENDIF
187  x2=log((1+(x**2.))/2.)
188  psym=(2.*log((1+x)/2.))+x2-(2.*atan(x))+piover2
189  ENDIF
190 
191  ELSEIF(zl>neut_limit) THEN !Stable
192 
193  IF(stabilitymethod==1)THEN !Dyer (1974) k=0.35 x=1+5*zl Mod. Hogstrom (1988)
194  psym=(-4.8)*zl_f
195  ELSEIF(stabilitymethod==2)THEN !Van Ulden & Holtslag (1985) p 1206
196  IF ( zl_f >1000. ) THEN
197  zl_f=1000.
198  END IF
199  psym=(-17.*(1.-exp(-0.29*zl_f)))
200  ELSEIF(stabilitymethod==4)THEN ! Businger et al (1971) modifed Hogstrom (1988)
201  ! psym=1+6*zl_f ! this is NOT the integral form but the stability function, TS 13 Jun 2017
202  psym=(-6)*zl_f ! this is the integral form, TS 13 Jun 2017
203  ELSEIF(stabilitymethod==3)THEN ! campbell & norman eqn 7.27 p 97
204  psym=(-6)*log(1+zl_f)
205 
206  ENDIF
207  ENDIF
208  RETURN
209 END FUNCTION stab_fn_mom

233

235  x=0.95*(1.-11.6*zl_f)**(-0.5)
236  ELSEIF(stabilitymethod==7) THEN
237  x=(1-(28.*zl))**0.25

239  x=0.95*(1.-15.2*zl_f)**0.5
240  ENDIF
241  psyh=2*log((1+x**2)/2)
242  ENDIF
243 
244  ELSE IF (zl>neut_limit) THEN !Stable
245  IF ( zl<=1 ) THEN ! weak/moderate stable
246  IF(stabilitymethod==4)THEN !Businger et al (1971) modifed Hogstrom (1988)
247  ! psyh=0.95+(7.8*zl_f) ! this is NOT the integral form but the stability function, TS 13 Jun 2017
248  psyh=(-7.8)*zl_f ! this is the integral form, TS 13 Jun 2017
249  ELSE !Dyer (1974) PSYH=(-5)*ZL modifed Hogstrom (1988)
250  psyh=(-4.5)*zl_f
251  ENDIF
252  ELSE !zL>1, very stable. otherwise psyh would be too large. TS 13 Jun 2017
253  ! adopt the form as Brutasert (1982) eqn 4.58. but following the coeffs. of the above eqns
254  IF(stabilitymethod==4)THEN !Businger et al (1971) modifed Hogstrom (1988)
255  psyh=(-7.8)*(1+log(zl_f))
256  ELSE !Dyer (1974) PSYH=(-5)*ZL modifed Hogstrom (1988)
257  psyh=(-4.5)*(1+log(zl_f))
258  ENDIF
259  END IF
260 
261  ENDIF
262 
263  RETURN
