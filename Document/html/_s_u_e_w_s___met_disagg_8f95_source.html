548  INTEGER:: Nper_loc !Number of subintervals per interval (local Nper)

552  REAL(KIND(1d0)),DIMENSION(ReadLinesOrig_loc):: Slow !Array to disaggregate
553  REAL(KIND(1d0)):: SlowPrev, SlowNext
554  INTEGER,DIMENSION(Nper_loc):: FastRows !Group of rows that are filled with each iteration
555  INTEGER,DIMENSION(FLOOR(Nper_loc/2.0)):: FirstRows10 !Rows at the beginning that are not filled during iteration (for averages)
556  INTEGER,DIMENSION(Nper_loc-FLOOR(Nper_loc/2.0)):: LastRows10 !Rows at the end that are not filled during iteration
557  INTEGER,DIMENSION(Nper_loc):: FirstRows20 !Rows at the beginning that are not filled during iteration (for instantaneous)

562  ! Calculate XNper_loc (differentiates between disaggregations with odd and even Nper_loc)
563  IF(mod(nper_loc,2)==0) xnper_loc=2
564  IF(mod(nper_loc,2)==1) xnper_loc=1
565 
566  seq1nper_loc = (/(i, i=1,nper_loc, 1)/)
567 
568  ! Setup counters for iteration
569  IF(disaggtype==10) THEN
570  fastrows = floor(nper_loc/2.0) + seq1nper_loc ! Rows to create at model time-step
571  firstrows10 = (/(i, i=1,(fastrows(1)-1), 1)/) !For start of dataset
572  lastrows10 = (/(i, i=nper_loc*(readlinesorigmax_loc-1-1)+fastrows(nper_loc)+1,(readlinesorigmax_loc*nper_loc),1)/) ! For end of dataset
573  ELSEIF(disaggtype==20) THEN
574  fastrows = nper_loc + seq1nper_loc !Rows to create at model time-step
575  firstrows20 = (/(i, i=1,(fastrows(1)-1), 1)/) !For start of dataset
576  ENDIF
577 
578  ! Initialise fast array to -999
579  fast = -999
580  ! Linearly disaggregate

582  IF(diagnosedisagg==1) write(*,*) 'Linearly disaggregating averaged variable'
583  DO i=1,(readlinesorigmax_loc-1)
584  fast(nper_loc*(i-1)+fastrows) = slow(i) - &
585  (slow(i+1)-slow(i))/(xnper_loc*nper_loc) + &
586  (slow(i+1)-slow(i))/nper_loc*(/(ii, ii=1,nper_loc, 1)/)
587  ENDDO
588 
589  ! For first few rows, use previous met block
590  IF(iblock==1) THEN
591  fast(firstrows10) = fast(fastrows(1)) !Use repeat values at the start of the year
592  ELSE
593  fast(firstrows10) = slowprev - &
594  (slow(1)-slowprev)/(xnper_loc*nper_loc) + &
595  (slow(1)-slowprev)/nper_loc * &
596  (/(ii, ii=(nper_loc-SIZE(firstrows10)+1),nper_loc, 1)/)
597  ENDIF

603  (slownext-slow(readlinesorigmax_loc))/(xnper_loc*nper_loc) + &
604  (slownext-slow(readlinesorigmax_loc))/nper_loc * &
605  (/(ii, ii=1,SIZE(lastrows10), 1)/)
606  ENDIF
607  ELSEIF(disaggtype==20) THEN !Instantaneous variables
608  IF(diagnosedisagg==1) write(*,*) 'Linearly disaggregating instantaneous variable'
609  DO i=1,(readlinesorigmax_loc-1)
610  fast(nper_loc*(i-1)+fastrows) = (slow(i) + &
611  (slow(i+1)-slow(i))/nper_loc*2*(seq1nper_loc-1) + &
612  slow(i))/2
613  ENDDO
614  ! For first few rows, use previous met block
615  IF(iblock==1) THEN
616  fast(firstrows20) = fast(fastrows(1)) !Use repeat values at the start of the year
617  ELSE
618  fast(firstrows20) = (slowprev + &
619  (slow(1)-slowprev)/nper_loc*2 * &
620  ((/(ii, ii=(nper_loc-SIZE(firstrows20)+1),nper_loc, 1)/)-1) + &
621  slowprev)/2
622  ENDIF
623  !! Last few rows are already filled for the instantaneous value disaggregation
624  !IF(iBlock==ReadBlocksOrigMetData) THEN

626  !ELSE
627  ! Fast(LastRows20) = (Slow(ReadLinesOrigMax_loc) + &

629  ! ((/(ii, ii=1,SIZE(LastRows20), 1)/)-1) + &
630  ! Slow(ReadLinesOrigMax_loc))/2
631  !ENDIF
632  ENDIF

643 FUNCTION disaggp_amongn(Slow,amongN, Nper_loc, ReadLinesOrig_loc, ReadLinesOrigMax_loc) RESULT(Fast)
644 ! Subroutine to disaggregate precipitation by evenly distributing among N subintervals
645 ! (i.e. equal intensity in N subintervals)
646 ! See Ward et al. (in review), meanN, 0.5N or 0.25N approach
647 ! HCW 10 Feb 2017
648 !======================================================================================
649 
650  USE defaultnotused
651  USE sues_data
652 
653  IMPLICIT NONE
654 
655  INTEGER:: amongN !Number of subintervals over which rain will be distributed
656  INTEGER:: Nper_loc !Number of subintervals per interval (local Nper)
657  INTEGER:: ReadLinesOrig_loc,ReadLinesOrigMax_loc !Number of lines to read in original file (local)
658  REAL(KIND(1d0)),DIMENSION(ReadLinesOrig_loc*Nper_loc):: Fast !Array to receive disaggregated data
659  REAL(KIND(1d0)),DIMENSION(ReadLinesOrig_loc):: Slow !Array to disaggregate
660  INTEGER,DIMENSION(:),ALLOCATABLE:: Subintervals !Array of subintervals that contain rain
661  INTEGER,DIMENSION(Nper_loc):: seq1Nper_loc !1 to Nper_loc
662  INTEGER:: i
663 
664  ! For each averaging period, get subintervals which will receive rain
665  ALLOCATE(subintervals(amongn))
666  subintervals(:) = -999
667 
668  seq1nper_loc = (/(i, i=1,nper_loc, 1)/)
669 
670  IF(diagnosedisagg==1) write(*,*) 'Distributing over ',amongn,' subintervals for variable'
671  ! If all subintervals are to contain rain, don't need to generate random numbers
672  IF(amongn == nper_loc) THEN
673  subintervals(:) = seq1nper_loc
674  ENDIF
675  IF(amongn > nper_loc) CALL errorhint(2,'Problem in SUEWS_MetDisagg: no. of rainy periods cannot exceed number of subintervals', &
676  REAL(Nper_loc,KIND(1d0)),NotUsed,amongN)

678

682  fast(nper_loc*(i-1)+seq1nper_loc) = 0 !Fill all subintervals with zeros initially
683  IF(slow(i) > 0) THEN !If there is some rainfall during this interval...
684  IF(amongn < nper_loc) THEN
685  subintervals(:) = -999

687  ENDIF
688  fast(nper_loc*(i-1)+subintervals) = slow(i)/amongn
689  ENDIF
690  ENDDO

692  IF(any(fast(1:readlinesorigmax_loc*nper_loc) == -999)) THEN
693  write(*,*) 'Problem: -999s (',count(fast(1:readlinesorigmax_loc*nper_loc) == -999),') in disaggregated data'
