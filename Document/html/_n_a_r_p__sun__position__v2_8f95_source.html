27  ! This function compute the sun position (zenith and azimuth angle (in degrees) at the observer
28  ! location) as a function of the observer local time and position.
29  !
30  ! Input lat and lng should be in degrees, alt in meters.
31  !
32  ! It is an implementation of the algorithm presented by Reda et Andreas in:
33  ! Reda, I., Andreas, A. (2003) Solar position algorithm for solar
34  ! radiation application. National Renewable Energy Laboratory (NREL)
35  ! Technical report NREL/TP-560-34302.
36  ! This document is avalaible at www.osti.gov/bridge
37  ! Code is translated from matlab code by Fredrik Lindberg (fredrikl@gvc.gu.se)
38  ! Last modified: LJ 27 Jan 2016 - Tabs removed

41  ! Convert to timevectors from dectime and year
42  call dectime_to_timevec(idectime,hour,min,sec)
43  dayofyear=floor(idectime)
44  call day2month(dayofyear,month,day,seas,year,locationlatitude)
45 
46  ! 1. Calculate the Julian Day, and Century. Julian Ephemeris day, century
47  ! and millenium are calculated using a mean delta_t of 33.184 seconds.
48  call julian_calculation(year,month,day,hour,min,sec,utc,juliancentury,julianday,julianephemeris_century,&
49  &julianephemeris_day,julianephemeris_millenium)
50 
51  ! 2. Calculate the Earth heliocentric longitude, latitude, and radius
52  ! vector (L, B, and R)
53  call earth_heliocentric_position_calculation(julianephemeris_millenium,earth_heliocentric_positionlatitude,&
54  &earth_heliocentric_positionlongitude,earth_heliocentric_positionradius)
55 
56  ! 3. Calculate the geocentric longitude and latitude
57  call sun_geocentric_position_calculation(earth_heliocentric_positionlongitude,earth_heliocentric_positionlatitude,&
58  & sun_geocentric_positionlatitude, sun_geocentric_positionlongitude)
59 
60  ! 4. Calculate the nutation in longitude and obliquity (in degrees).
61  call nutation_calculation(julianephemeris_century,nutationlongitude,nutationobliquity)
62 
63  ! 5. Calculate the true obliquity of the ecliptic (in degrees).
64  call corr_obliquity_calculation(julianephemeris_millenium, nutationobliquity, corr_obliquity)
65 
66  ! 6. Calculate the aberration correction (in degrees)
67  call abberation_correction_calculation(earth_heliocentric_positionradius, aberration_correction)
68 
69  ! 7. Calculate the apparent sun longitude in degrees)
70  call apparent_sun_longitude_calculation(sun_geocentric_positionlongitude, nutationlongitude,&
71  & aberration_correction, apparent_sun_longitude)
72 
73  ! 8. Calculate the apparent sideral time at Greenwich (in degrees)
74  call apparent_stime_at_greenwich_calculation(julianday,juliancentury, nutationlongitude, &
75  &corr_obliquity, apparent_stime_at_greenwich)
76 
77  ! 9. Calculate the sun rigth ascension (in degrees)
78  call sun_rigth_ascension_calculation(apparent_sun_longitude, corr_obliquity, sun_geocentric_positionlatitude, &
79  &sun_rigth_ascension)
80 
81  ! 10. Calculate the geocentric sun declination (in degrees). Positive or
82  ! negative if the sun is north or south of the celestial equator.
83  call sun_geocentric_declination_calculation(apparent_sun_longitude, corr_obliquity, sun_geocentric_positionlatitude, &
84  &sun_geocentric_declination)
85 
86  ! 11. Calculate the observer local hour angle (in degrees, westward from south).
87  call observer_local_hour_calculation(apparent_stime_at_greenwich, locationlongitude, sun_rigth_ascension, observer_local_hour)
88 
89  ! 12. Calculate the topocentric sun position (rigth ascension, declination and
90  ! rigth ascension parallax in degrees)
91  call topocentric_sun_position_calculate(topocentric_sun_positionrigth_ascension,&
92  &topocentric_sun_positionrigth_ascension_parallax,topocentric_sun_positiondeclination,locationaltitude,&
93  &locationlatitude,observer_local_hour,sun_rigth_ascension,sun_geocentric_declination,&
94  &earth_heliocentric_positionradius)
95 
96  ! 13. Calculate the topocentric local hour angle (in degrees)
97  call topocentric_local_hour_calculate(observer_local_hour, topocentric_sun_positionrigth_ascension_parallax,&
98  & topocentric_local_hour)
99 
100  ! 14. Calculate the topocentric zenith and azimuth angle (in degrees)
101  call sun_topocentric_zenith_angle_calculate(locationlatitude , topocentric_sun_positiondeclination,&
102  & topocentric_local_hour, sunazimuth,sunzenith)

110 subroutine julian_calculation(year,month,day,hour,min,sec,UTC,juliancentury,julianday,julianephemeris_century&
111  &,julianephemeris_day,julianephemeris_millenium)
112  implicit none
113 
114  REAL(KIND(1D0)) :: A,B,D,delta_t
115  REAL(KIND(1D0)) :: juliancentury
116  REAL(KIND(1D0)) :: julianday
117  REAL(KIND(1D0)) :: julianephemeris_century
118  REAL(KIND(1D0)) :: julianephemeris_day
119  REAL(KIND(1D0)) :: julianephemeris_millenium
120  REAL(KIND(1D0)) :: M,sec,year,UTC
121  integer :: day,hour,min,month
122  !REAL(KIND(1D0)) :: time !>
123  REAL(KIND(1D0)) :: ut_time ,Y !tt,
124  !
125  ! This function compute the julian day and julian century from the local
126  ! time and timezone information. Ephemeris are calculated with a delta_t=0
127  ! seconds.
128 
129  if (month == 1 .or. month == 2) then
130  y = year - 1.
131  m = month + 12
132  else
133  y = year
134  m = month
135  end if
136  ut_time = ((float(hour) - utc)/24.) + (float(min)/(60.*24.)) + (sec/(60.*60.*24.)) ! time of day in UT time.
137  d = day + ut_time ! Day of month in decimal time, ex. 2sd day of month at 12:30:30UT, D=2.521180556
138 
139  ! In 1582, the gregorian calendar was adopted
140  if (year == 1582.) then
141  if (month == 10) then
142  if (day <= 4) then ! The Julian calendar ended on October 4, 1582
143  b = 0
144  else if (day >= 15) then ! The Gregorian calendar started on October 15, 1582
145  a = floor(y/100)
146  b = 2 - a + floor(a/4)
147  else
148  !disp('This date never existed!. Date automatically set to October 4, 1582')
149  month = 10
150  day = 4
151  b = 0
152  end if
153  else if (month<10) then ! Julian calendar
154  b = 0
155  else ! Gregorian calendar
156  a = floor(y/100)
157  b = 2 - a + floor(a/4)
158  end if
159 
160  else if (year<1582.) then ! Julian calendar
161  b = 0
162  else
163  a = floor(y/100) ! Gregorian calendar
164  b = 2 - a + floor(a/4)
165  end if
166 
167  julianday = floor(365.25*(y+4716.)) + floor(30.6001*(m+1)) + d + b - 1524.5
168 
169  delta_t = 0. ! 33.184;
170  julianephemeris_day = julianday + (delta_t/86400)
171 
172  juliancentury = (julianday - 2451545.) / 36525.
173 
174  julianephemeris_century = (julianephemeris_day - 2451545.) / 36525.
175 
176  julianephemeris_millenium = julianephemeris_century / 10.
177 
178 end subroutine julian_calculation
