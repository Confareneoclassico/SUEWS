201  ! OHM coefficients --------
202  ! Set to zero initially
203  a1=0 ![-]
204  a2=0 ![h]
205  a3=0 ![W m-2]
206  ! -------------------------
207 
208  ! Loop through surface types ----------------------------------------------------------
209  DO is=1,nsurf
210  surfrac=sfr(is)
211 
212  ! Use 5-day running mean Tair to decide whether it is summer or winter ----------------
213  IF(hddday >= ohm_threshsw(is)) THEN !Summer
214  ii=0
215  ELSE !Winter
216  ii=2
217  ENDIF
218 
219  IF(state(is) > 0) THEN !Wet surface
220  i=ii+1
221  ELSE !Dry surface
222  i=ii+2
223  ! If the surface is dry but SM is close to capacity, use coefficients for wet surfaces
224  IF(is>bldgsurf.AND.is/=watersurf)THEN !Wet soil (i.e. EveTr, DecTr, Grass, BSoil surfaces)
225  IF(soilmoist(is)/soilstorecap(is) > ohm_threshwd(is) ) THEN
226  i=ii+1
227  ENDIF
228  ENDIF
229  ENDIF
230 
231  ! If snow, adjust surface fractions accordingly
232  IF(snowuse==1.AND.is/=bldgsurf.AND.is/=watersurf) THEN ! ?? Why is BldgSurf excluded here?
233  surfrac=surfrac*(1-snowfrac(is))
234  ENDIF
235 
236  ! Calculate the areally-weighted OHM coefficients
237  a1 = a1+surfrac*ohm_coef(is,i,1)
238  a2 = a2+surfrac*ohm_coef(is,i,2)
