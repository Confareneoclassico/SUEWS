466  FUNCTION transmissivity(Press_hPa,Temp_C_dew,G,zenith) RESULT(trans)
467  ! bulk atmospheric transmissivity (Crawford and Duchon, 1999)
468  ! P = pressure (hPa)
469  ! Td = dewpoint (C)
470  ! G parameter is empirical value from Smith 1966 (JAM)
471  ! zenith in radians
472  ! if zenith > 80 use the value for 80.
473 
474  REAL(KIND(1d0)) ::Press_hPa,TemP_C_dew,zenith,G,trans
475  REAL(KIND(1d0))::m,TrTpg,u,Tw,Ta,cosZ
476  REAL(KIND(1d0))::Tdf
477 
478  IF (zenith>80.*deg2rad) THEN
479  cosz=cos(80.*deg2rad)
480  ELSE
481  cosz=cos(zenith)
482  ENDIF
483 
484  tdf = temp_c_dew*1.8+32. !celsius to fahrenheit
485  ! Transmission coefficients
486  m = 35*cosz/sqrt(1224.*cosz*cosz+1) !optical air mass at p=1013 mb
487  !Rayleigh & permanent gases
488  trtpg = 1.021-0.084*sqrt(m*(0.000949*press_hpa+0.051)) !first two trans coeff
489  u = exp(0.113-log(g+1)+0.0393*tdf) !precipitable water
490  tw = 1-0.077*(u*m)**0.3 !vapor transmission coe3ff.
491  ta = 0.935**m !4th trans coeff
492  trans = trtpg*tw*ta !bulk atmospheric transmissivity
493  END FUNCTION transmissivity
